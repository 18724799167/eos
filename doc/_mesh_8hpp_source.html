61 inline void write_obj(Mesh mesh, std::string filename)
62 {
63  assert(mesh.vertices.size() == mesh.colors.size() || mesh.colors.empty());
64

80  << mesh.colors[i][2] << std::endl;
81  }
82  }
83 
84  if (!mesh.texcoords.empty())
85  {
86  for (auto&& tc : mesh.texcoords)
87  {
88  obj_file << "vt " << tc[0] << " " << tc[1] << std::endl;
89  }
90  }
91 
92  for (auto&& v : mesh.tvi)
93  {
94  // Add one because obj starts counting triangle indices at 1
95  obj_file << "f " << v[0] + 1 << " " << v[1] + 1 << " " << v[2] + 1 << std::endl;
96  }
97 
98  return;

112 inline void write_textured_obj(Mesh mesh, std::string filename)
113 {
114  assert((mesh.vertices.size() == mesh.colors.size() || mesh.colors.empty()) && !mesh.texcoords.empty());
115 
116  if (filename.at(filename.size() - 4) != '.')
117  {
118  throw std::runtime_error(
119  "Error in given filename: Expected a dot and a 3-letter extension at the end (i.e. '.obj'). " +
120  filename);
121  }
122 
123  // Takes a full path to a file and returns only the filename:
124  const auto get_filename = [](const std::string& path) {
125  auto last_slash = path.find_last_of("/\\");
126  if (last_slash == std::string::npos)
127  {
128  return path;
129  }
130  return path.substr(last_slash + 1, path.size());
131  };
132 
133  std::ofstream obj_file(filename);
134 
135  std::string mtl_filename(filename);
136  // replace '.obj' at the end with '.mtl':
137  mtl_filename.replace(std::end(mtl_filename) - 4, std::end(mtl_filename), ".mtl");
138 
139  obj_file << "mtllib " << get_filename(mtl_filename) << std::endl; // first line of the obj file
140 
141  // same as in write_obj():
142  if (mesh.colors.empty())
143  {
144  for (std::size_t i = 0; i < mesh.vertices.size(); ++i)
145  {
146  obj_file << "v " << mesh.vertices[i][0] << " " << mesh.vertices[i][1] << " "
147  << mesh.vertices[i][2] << " " << std::endl;
148  }
149  } else
150  {
151  for (std::size_t i = 0; i < mesh.vertices.size(); ++i)
152  {
153  obj_file << "v " << mesh.vertices[i][0] << " " << mesh.vertices[i][1] << " "
154  << mesh.vertices[i][2] << " " << mesh.colors[i][0] << " " << mesh.colors[i][1] << " "
155  << mesh.colors[i][2] << " " << std::endl;
156  }
157  }
158  // end
159 
160  for (std::size_t i = 0; i < mesh.texcoords.size(); ++i)
161  {
162  obj_file << "vt " << mesh.texcoords[i][0] << " " << 1.0f - mesh.texcoords[i][1] << std::endl;
163  // We invert y because Meshlab's uv origin (0, 0) is on the bottom-left
164  }
165 
166  obj_file << "usemtl FaceTexture" << std::endl; // the name of our texture (material) will be 'FaceTexture'
167 
168  for (auto&& v : mesh.tvi)
169  {
170  // This assumes mesh.texcoords.size() == mesh.vertices.size(). The texture indices could theoretically be different (for example in the cube-mapped 3D scan).
171  // Add one because obj starts counting triangle indices at 1
172  obj_file << "f " << v[0] + 1 << "/" << v[0] + 1 << " " << v[1] + 1 << "/" << v[1] + 1 << " "
173  << v[2] + 1 << "/" << v[2] + 1 << std::endl;
174  }
175 
176  std::ofstream mtl_file(mtl_filename);
177  std::string texture_filename(filename);
178  // replace '.obj' at the end with '.isomap.png':
179  texture_filename.replace(std::end(texture_filename) - 4, std::end(texture_filename), ".isomap.png");
180 
181  mtl_file << "newmtl FaceTexture" << std::endl;
182  mtl_file << "map_Kd " << get_filename(texture_filename) << std::endl;
183 
184  return;
185 };
