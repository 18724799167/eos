67  template <typename T, glm::precision P = glm::defaultp>
68  void raster_triangle(const detail::Vertex<T, P>& point_a, const detail::Vertex<T, P>& point_b,
69  const detail::Vertex<T, P>& point_c, const cpp17::optional<Texture>& texture)
70  {
71  // We already calculated this in the culling/clipping stage. Maybe we should save/cache it after all.
72  const auto boundingBox = detail::calculate_clipped_bounding_box(
73  glm::tvec2<T, P>(point_a.position.x, point_a.position.y),
74  glm::tvec2<T, P>(point_b.position.x, point_b.position.y),
75  glm::tvec2<T, P>(point_c.position.x, point_c.position.y), viewport_width, viewport_height);
76  const auto min_x = boundingBox.x;
77  const auto max_x = boundingBox.x + boundingBox.width;
78  const auto min_y = boundingBox.y;
79  const auto max_y = boundingBox.y + boundingBox.height;
80 
81  // These are triangle-specific, i.e. calculate once per triangle.
82  // These ones are needed for perspective correct lambdas! (as well as mipmapping)
83  const auto& one_over_w0 = point_a.position[3];
84  const auto& one_over_w1 = point_b.position[3];
85  const auto& one_over_w2 = point_c.position[3];
86 
87  // These are triangle-specific, i.e. calculate once per triangle.
88  // For partial derivatives computation (for mipmapping, texturing) (they work on screen-space coords):
89  using eos::render::detail::plane;
90  const auto alpha_plane = plane(
91  glm::tvec3<T, P>(point_a.position[0], point_a.position[1], point_a.texcoords[0] * one_over_w0),
92  glm::tvec3<T, P>(point_b.position[0], point_b.position[1], point_b.texcoords[0] * one_over_w1),
93  glm::tvec3<T, P>(point_c.position[0], point_c.position[1], point_c.texcoords[0] * one_over_w2));
94  const auto beta_plane = plane(
95  glm::tvec3<T, P>(point_a.position[0], point_a.position[1], point_a.texcoords[1] * one_over_w0),
96  glm::tvec3<T, P>(point_b.position[0], point_b.position[1], point_b.texcoords[1] * one_over_w1),
97  glm::tvec3<T, P>(point_c.position[0], point_c.position[1], point_c.texcoords[1] * one_over_w2));
98  const auto gamma_plane =
99  plane(glm::tvec3<T, P>(point_a.position[0], point_a.position[1], one_over_w0),
100  glm::tvec3<T, P>(point_b.position[0], point_b.position[1], one_over_w1),
101  glm::tvec3<T, P>(point_c.position[0], point_c.position[1], one_over_w2));
102  const auto one_over_alpha_c = 1.0f / alpha_plane.c;
103  const auto one_over_beta_c = 1.0f / beta_plane.c;
104  const auto one_over_gamma_c = 1.0f / gamma_plane.c;
105  const auto alpha_ffx = -alpha_plane.a * one_over_alpha_c;
106  const auto beta_ffx = -beta_plane.a * one_over_beta_c;
107  const auto gamma_ffx = -gamma_plane.a * one_over_gamma_c;
108  const auto alpha_ffy = -alpha_plane.b * one_over_alpha_c;
109  const auto beta_ffy = -beta_plane.b * one_over_beta_c;
110  const auto gamma_ffy = -gamma_plane.b * one_over_gamma_c;
111 
112  for (int yi = min_y; yi <= max_y; ++yi)
113  {
114  for (int xi = min_x; xi <= max_x; ++xi)
115  {
116  // we want centers of pixels to be used in computations. Todo: Do we? Do we pass it with or
117  // without +0.5 to the FragShader?
118  const float x = static_cast<float>(xi) + 0.5f; // double? T?
119  const float y = static_cast<float>(yi) + 0.5f;
120 
121  // These will be used for barycentric weights computation
122  using detail::implicit_line;
123  const double one_over_v0ToLine12 =
124  1.0 / implicit_line(point_a.position[0], point_a.position[1], point_b.position,
125  point_c.position);
126  const double one_over_v1ToLine20 =
127  1.0 / implicit_line(point_b.position[0], point_b.position[1], point_c.position,
128  point_a.position);
129  const double one_over_v2ToLine01 =
130  1.0 / implicit_line(point_c.position[0], point_c.position[1], point_a.position,
131  point_b.position);
132  // Affine barycentric weights:
133  double alpha = implicit_line(x, y, point_b.position, point_c.position) * one_over_v0ToLine12;
134  double beta = implicit_line(x, y, point_c.position, point_a.position) * one_over_v1ToLine20;
135  double gamma = implicit_line(x, y, point_a.position, point_b.position) * one_over_v2ToLine01;
136 
137  // if pixel (x, y) is inside the triangle or on one of its edges
138  if (alpha >= 0 && beta >= 0 && gamma >= 0)
139  {
140  const int pixel_index_row = yi;
141  const int pixel_index_col = xi;
142 
143  // TODO: Check this one. What about perspective?
144  const double z_affine = alpha * static_cast<double>(point_a.position[2]) +
145  beta * static_cast<double>(point_b.position[2]) +
146  gamma * static_cast<double>(point_c.position[2]);
147 
148  bool draw = true;
149  if (enable_far_clipping)
150  {
151  if (z_affine > 1.0)
152  {
153  draw = false;
154  }
155  }
156 
157  bool passes_depth_test = false;
158  if (enable_depth_test)
159  {
160  // If enable_depth_test=false, avoid accessing the depthbuffer at all - it might be
161  // empty or have other dimensions.
162  passes_depth_test =
163  (z_affine < depthbuffer(pixel_index_row, pixel_index_col));
164  }
165  // The '<= 1.0' clips against the far-plane in NDC. We clip against the near-plane
166  // earlier.
167  // if (z_affine < depthbuffer.at<double>(pixelIndexRow, pixelIndexCol)/* && z_affine <=
168  // 1.0*/) // what to do in ortho case without n/f "squashing"? should we always squash? or
169  // a flag?
170  if ((passes_depth_test && draw) || enable_depth_test == false)
171  {
172  // perspective-correct barycentric weights
173  // Todo: Check this in the original/older implementation, i.e. if all is still
174  // perspective-correct. I think so. Also compare 1:1 with OpenGL.
175  double d = alpha * one_over_w0 + beta * one_over_w1 + gamma * one_over_w2;
176  d = 1.0 / d;
177  if (!extracting_tex) // Pass the uncorrected lambda if we're extracting tex... hack...
178  // do properly!
179  {
180  alpha *= d * one_over_w0; // In case of affine cam matrix, everything is 1 and
181  // a/b/g don't get changed.
182  beta *= d * one_over_w1;
183  gamma *= d * one_over_w2;
184  }
185  glm::tvec3<T, P> lambda(alpha, beta, gamma);
186 
187  glm::tvec4<T, P> pixel_color;
188  if (texture)
189  {
190  // check if texture != NULL?
191  // partial derivatives (for mip-mapping, not needed for texturing otherwise!)
192  const float u_over_z =
193  -(alpha_plane.a * x + alpha_plane.b * y + alpha_plane.d) * one_over_alpha_c;
194  const float v_over_z =
195  -(beta_plane.a * x + beta_plane.b * y + beta_plane.d) * one_over_beta_c;
196  const float one_over_z =
197  -(gamma_plane.a * x + gamma_plane.b * y + gamma_plane.d) * one_over_gamma_c;
198  const float one_over_squared_one_over_z = 1.0f / std::pow(one_over_z, 2);
199 
200  // partial derivatives of U/V coordinates with respect to X/Y pixel's screen
201  // coordinates
202  // These are exclusively used for the mipmap level computation (i.e. which mipmap
203  // levels to use).
204  // They're not needed for texturing otherwise at all!
205  float dudx =
206  one_over_squared_one_over_z * (alpha_ffx * one_over_z - u_over_z * gamma_ffx);
207  float dudy =
208  one_over_squared_one_over_z * (beta_ffx * one_over_z - v_over_z * gamma_ffx);
209  float dvdx =
210  one_over_squared_one_over_z * (alpha_ffy * one_over_z - u_over_z * gamma_ffy);
211  float dvdy =
212  one_over_squared_one_over_z * (beta_ffy * one_over_z - v_over_z * gamma_ffy);
213  dudx *= texture.value().mipmaps[0].width();
214  dudy *= texture.value().mipmaps[0].width();
215  dvdx *= texture.value().mipmaps[0].height();
216  dvdy *= texture.value().mipmaps[0].height();
217 
218  // Why does it need x and y? Maybe some shaders (eg TexExtr?) need it?
219  pixel_color = fragment_shader.shade_triangle_pixel(
220  x, y, point_a, point_b, point_c, lambda, texture, dudx, dudy, dvdx, dvdy);
221 
222  } else
223  { // We use vertex-coloring
224  // Why does it need x and y?
225  pixel_color = fragment_shader.shade_triangle_pixel(
226  x, y, point_a, point_b, point_c, lambda, texture, 0, 0, 0, 0);
227  }
228 
229  // clamp bytes to 255
230  // Todo: Proper casting (rounding?)? And we don't clamp/max against 255? Use
231  // glm::clamp?
232  const unsigned char red =
233  static_cast<unsigned char>(255.0f * std::min(pixel_color[0], T(1)));
234  const unsigned char green =
235  static_cast<unsigned char>(255.0f * std::min(pixel_color[1], T(1)));
236  const unsigned char blue =
237  static_cast<unsigned char>(255.0f * std::min(pixel_color[2], T(1)));
238  const unsigned char alpha =
239  static_cast<unsigned char>(255.0f * std::min(pixel_color[3], T(1)));
240 
241  // update buffers
242  colorbuffer(pixel_index_row, pixel_index_col)[0] = blue;
243  colorbuffer(pixel_index_row, pixel_index_col)[1] = green;
244  colorbuffer(pixel_index_row, pixel_index_col)[2] = red;
245  colorbuffer(pixel_index_row, pixel_index_col)[3] = alpha;
246  if (enable_depth_test) // TODO: A better name for this might be enable_zbuffer? or
247  // enable_zbuffer_test?
248  {
249  depthbuffer(pixel_index_row, pixel_index_col) = z_affine;
250  }
251  }
252  }
253  }
254  }
255  };
256

258  FragmentShaderType fragment_shader;
259

262  bool extracting_tex = false;
263  bool enable_far_clipping = true;
264

266  int viewport_height;
267

269  eos::core::Image1d depthbuffer;
270 };
