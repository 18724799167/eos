75  bool try_add(float frame_score, const ImageType& image, const fitting::FittingResult& fitting_result)
76  {
77  // Determine whether to add or not:
78  auto yaw_angle = glm::degrees(glm::yaw(fitting_result.rendering_parameters.get_rotation()));
79  auto idx = angle_to_index(yaw_angle);
80  bool add_frame = false;
81  if (bins[idx].size() < frames_per_bin) // always add when we don't have enough frames
82  add_frame =
83  true; // definitely adding - we wouldn't have to go through the for-loop on the next line.
84  for (auto&& f : bins[idx])
85  {
86  if (frame_score > f.score)
87  add_frame = true;
88  }
89  if (!add_frame)
90  {
91  return false;
92  }
93  // Add the keyframe:
94  bins[idx].push_back(video::Keyframe{frame_score, image, fitting_result});
95  if (bins[idx].size() > frames_per_bin)
96  {
97  // need to remove the lowest one:
98  std::sort(std::begin(bins[idx]), std::end(bins[idx]),
99  [](const auto& lhs, const auto& rhs) { return lhs.score > rhs.score; });
100  bins[idx].resize(frames_per_bin);
101  }
102  return true;
103  };
