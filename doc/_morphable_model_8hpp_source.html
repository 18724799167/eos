138  core::Mesh get_mean() const
139  {
140  assert(shape_model.get_data_dimension() == color_model.get_data_dimension() ||
141  !has_color_model()); // The number of vertices (= model.getDataDimension() / 3) has to be equal
142  // for both models, or, alternatively, it has to be a shape-only model.
143 
144  Eigen::VectorXf shape = shape_model.get_mean();
145  const Eigen::VectorXf color = color_model.get_mean();
146 
147  // If there is a PCA expression model, add that model's mean. If there is no expression model, or
148  // blendshapes, there's nothing to add.
149  if (get_expression_model_type() == ExpressionModelType::PcaModel)
150  {
151  shape += cpp17::get<PcaModel>(expression_model.value()).get_mean();
152  }
153 
154  core::Mesh mesh;
155  if (has_texture_coordinates())
156  {
157  mesh = sample_to_mesh(shape, color, shape_model.get_triangle_list(),
158  color_model.get_triangle_list(), texture_coordinates);
159  } else
160  {
161  mesh = sample_to_mesh(shape, color, shape_model.get_triangle_list(),
162  color_model.get_triangle_list());
163  }
164  return mesh;

269  core::Mesh draw_sample(std::vector<float> shape_coefficients, std::vector<float> expression_coefficients, std::vector<float> color_coefficients) const
270  {
271  assert(shape_model.get_data_dimension() == color_model.get_data_dimension() ||
272  !has_color_model()); // The number of vertices (= model.getDataDimension() / 3) has to be equal
273  // for both models, or, alternatively, it has to be a shape-only model.
274  assert(has_separate_expression_model());
275 
276  Eigen::VectorXf shape_sample;
277  Eigen::VectorXf expression_sample;
278  Eigen::VectorXf color_sample;
279 
280  if (shape_coefficients.empty())
281  {
282  shape_sample = shape_model.get_mean();
283  } else
284  {
285  shape_sample = shape_model.draw_sample(shape_coefficients);
286  }
287  // Get a sample of the expression model, depending on whether it's a PcaModel or Blendshapes:
288  if (get_expression_model_type() == ExpressionModelType::PcaModel) {
289  const auto& pca_expression_model = cpp17::get<PcaModel>(expression_model.value());
290  assert(pca_expression_model.get_data_dimension() == shape_model.get_data_dimension());
291  if (expression_coefficients.empty())
292  {
293  expression_sample = pca_expression_model.get_mean();
294  } else
295  {
296  expression_sample = pca_expression_model.draw_sample(expression_coefficients);
297  }
298  } else if (get_expression_model_type() == ExpressionModelType::Blendshapes)
299  {
300  const auto& expression_blendshapes = cpp17::get<Blendshapes>(expression_model.value());
301  assert(expression_blendshapes.size() > 0);
302  assert(expression_blendshapes[0].deformation.rows() == shape_model.get_data_dimension());
303  if (expression_coefficients.empty())
304  {
305  expression_sample.setZero(expression_blendshapes[0].deformation.rows());
306  }
307  else
308  {
309  expression_sample = to_matrix(expression_blendshapes) * Eigen::Map<const Eigen::VectorXf>(expression_coefficients.data(), expression_coefficients.size());
310  }
311  } else
312  {
313  throw std::runtime_error("This MorphableModel doesn't contain an expression model in the form of a PcaModel or Blendshapes.");
314  }
315  shape_sample += expression_sample;
316 
317  if (color_coefficients.empty())
318  {
319  color_sample = color_model.get_mean();
320  } else
321  {
322  color_sample = color_model.draw_sample(color_coefficients);
323  }
324 
325  core::Mesh mesh;
326  if (has_texture_coordinates())
327  {
328  mesh = sample_to_mesh(shape_sample, color_sample, shape_model.get_triangle_list(),
329  color_model.get_triangle_list(), texture_coordinates);
330  } else
331  {
332  mesh = sample_to_mesh(shape_sample, color_sample, shape_model.get_triangle_list(),
333  color_model.get_triangle_list());
334  }
335  return mesh;

377  ExpressionModelType get_expression_model_type() const
378  {
379  if (!expression_model)
380  {
381  return ExpressionModelType::None;
382  }
383  if (cpp17::holds_alternative<Blendshapes>(expression_model.value()))
384  {
385  return ExpressionModelType::Blendshapes;
386  } else if (cpp17::holds_alternative<PcaModel>(expression_model.value()))
387  {
388  return ExpressionModelType::PcaModel;
389  } else
390  {
391  // We should never get here - but this silences the "not all control paths return a value"
392  // compiler warning.
393  throw std::runtime_error("The ExpressionModel contains something, but it's not Blendshapes or a PcaModel. This should not happen.");
394  }

498 inline core::Mesh sample_to_mesh(const Eigen::VectorXf& shape_instance, const Eigen::VectorXf& color_instance,
499  const std::vector<std::array<int, 3>>& tvi,
500  const std::vector<std::array<int, 3>>& tci,
501  const std::vector<std::array<double, 2>>&
502  texture_coordinates /* = std::vector<std::array<double, 2>>() */)
503 {
504  assert(shape_instance.rows() == color_instance.rows() ||
505  color_instance.size() == 0); // The number of vertices (= model.getDataDimension() / 3) has to be
506  // equal for both models, or, alternatively, it has to be a shape-only
507  // model.
508 
509  const auto num_vertices = shape_instance.rows() / 3;
510 
511  core::Mesh mesh;
512 
513  // Construct the mesh vertices:
514  mesh.vertices.resize(num_vertices);
515  for (auto i = 0; i < num_vertices; ++i)
516  {
517  mesh.vertices[i] = Eigen::Vector3f(
518  shape_instance(i * 3 + 0), shape_instance(i * 3 + 1),
519  shape_instance(i * 3 + 2)); // Note: This can probably be simplified now, Eigen on both sides!
520  }
521 
522  // Assign the vertex colour information if it's not a shape-only model:
523  if (color_instance.size() > 0)
524  {
525  mesh.colors.resize(num_vertices);
526  for (auto i = 0; i < num_vertices; ++i)
527  {
528  mesh.colors[i] = Eigen::Vector3f(
529  cpp17::clamp(color_instance(i * 3 + 0), 0.0f, 1.0f),
530  cpp17::clamp(color_instance(i * 3 + 1), 0.0f, 1.0f),
531  cpp17::clamp(color_instance(i * 3 + 2), 0.0f, 1.0f)); // We use RGB order everywhere.
532  }
533  }
534 
535  // Assign the triangle lists:
536  mesh.tvi = tvi;
537  mesh.tci = tci; // tci will be empty in case of a shape-only model
538 
539  // Texture coordinates, if the model has them:
540  if (!texture_coordinates.empty())
541  {
542  mesh.texcoords.resize(num_vertices);
543  for (auto i = 0; i < num_vertices; ++i)
544  {
545  mesh.texcoords[i] = Eigen::Vector2f(
546  texture_coordinates[i][0],
547  texture_coordinates[i][1]); // Note: This can probably be simplified now, Eigen on both sides!
548  }
549  }
550 
551  return mesh;
552 };
