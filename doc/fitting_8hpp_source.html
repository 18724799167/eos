69 inline Eigen::VectorXf fit_shape(Eigen::Matrix<float, 3, 4> affine_camera_matrix,
70  const morphablemodel::MorphableModel& morphable_model,
71  const std::vector<morphablemodel::Blendshape>& blendshapes,
72  const std::vector<Eigen::Vector2f>& image_points,
73  const std::vector<int>& vertex_indices, float lambda,
74  cpp17::optional<int> num_coefficients_to_fit,
75  std::vector<float>& pca_shape_coefficients,
76  std::vector<float>& blendshape_coefficients)
77 {
78  using Eigen::MatrixXf;
79  using Eigen::VectorXf;
80 
81  const MatrixXf blendshapes_as_basis = morphablemodel::to_matrix(blendshapes);
82 
83  std::vector<float> last_blendshape_coeffs, current_blendshape_coeffs;
84  std::vector<float> last_pca_coeffs, current_pca_coeffs;
85  current_blendshape_coeffs.resize(blendshapes.size()); // starting values t_0, all zeros
86  // no starting values for current_pca_coeffs required, since we start with the shape fitting, and cv::norm
87  // of an empty vector is 0.
88 
89  VectorXf combined_shape;
90  do // run at least once:
91  {
92  last_blendshape_coeffs = current_blendshape_coeffs;
93  last_pca_coeffs = current_pca_coeffs;
94  // Estimate the PCA shape coefficients with the current blendshape coefficients (0 in the first
95  // iteration):
96  const VectorXf mean_plus_blendshapes =
97  morphable_model.get_shape_model().get_mean() +
98  blendshapes_as_basis *
99  Eigen::Map<const VectorXf>(last_blendshape_coeffs.data(), last_blendshape_coeffs.size());
100  current_pca_coeffs = fitting::fit_shape_to_landmarks_linear(
101  morphable_model.get_shape_model(), affine_camera_matrix, image_points, vertex_indices,
102  mean_plus_blendshapes, lambda, num_coefficients_to_fit);
103 
104  // Estimate the blendshape coefficients with the current PCA model estimate:
105  const VectorXf pca_model_shape = morphable_model.get_shape_model().draw_sample(current_pca_coeffs);
106  current_blendshape_coeffs = fitting::fit_blendshapes_to_landmarks_nnls(
107  blendshapes, pca_model_shape, affine_camera_matrix, image_points, vertex_indices);
108 
109  // Todo/Note: Could move next line outside the loop, not needed in here actually
110  combined_shape = pca_model_shape +
111  blendshapes_as_basis * Eigen::Map<const VectorXf>(current_blendshape_coeffs.data(),
112  current_blendshape_coeffs.size());
113  } while (
114  std::abs(Eigen::Map<const VectorXf>(current_pca_coeffs.data(), current_pca_coeffs.size()).norm() -
115  Eigen::Map<const VectorXf>(last_pca_coeffs.data(), last_pca_coeffs.size()).norm()) >= 0.01 ||
116  std::abs(
117  Eigen::Map<const VectorXf>(current_blendshape_coeffs.data(), current_blendshape_coeffs.size()).norm() -
118  Eigen::Map<const VectorXf>(last_blendshape_coeffs.data(), last_blendshape_coeffs.size()).norm()) >= 0.01);
119 
120  pca_shape_coefficients = current_pca_coeffs;
121  blendshape_coefficients = current_blendshape_coeffs;
122 
123  return combined_shape;
124 };

175 template <class T>
176 inline auto get_corresponding_pointset(const T& landmarks, const core::LandmarkMapper& landmark_mapper,
177  const morphablemodel::MorphableModel& morphable_model)
178 {
179  using Eigen::Vector2f;
180  using Eigen::Vector4f;
181  using std::vector;
182 
183  // These will be the final 2D and 3D points used for the fitting:
184  vector<Vector4f> model_points; // the points in the 3D shape model
185  vector<int> vertex_indices; // their vertex indices
186  vector<Vector2f> image_points; // the corresponding 2D landmark points
187 
188  // Sub-select all the landmarks which we have a mapping for (i.e. that are defined in the 3DMM):
189  for (int i = 0; i < landmarks.size(); ++i)
190  {
191  auto converted_name = landmark_mapper.convert(landmarks[i].name);
192  if (!converted_name)
193  { // no mapping defined for the current landmark
194  continue;
195  }
196  int vertex_idx = std::stoi(converted_name.get());
197  auto vertex = morphable_model.get_shape_model().get_mean_at_point(vertex_idx);
198  model_points.emplace_back(Vector4f(vertex.x(), vertex.y(), vertex.z(), 1.0f));
199  vertex_indices.emplace_back(vertex_idx);
200  image_points.emplace_back(landmarks[i].coordinates);
201  }
202  return std::make_tuple(image_points, model_points, vertex_indices);
203 };

215 template <class T>
216 inline auto concat(const std::vector<T>& vec_a, const std::vector<T>& vec_b)
217 {
218  std::vector<T> concatenated_vec;
219  concatenated_vec.reserve(vec_a.size() + vec_b.size());
220  concatenated_vec.insert(std::end(concatenated_vec), std::begin(vec_a), std::end(vec_a));
221  concatenated_vec.insert(std::end(concatenated_vec), std::begin(vec_b), std::end(vec_b));

243 inline std::vector<float>
244 fit_expressions(const morphablemodel::ExpressionModel& expression_model, const Eigen::VectorXf& face_instance,
245  const Eigen::Matrix<float, 3, 4>& affine_camera_matrix,
246  const std::vector<Eigen::Vector2f>& landmarks, const std::vector<int>& vertex_ids,
247  cpp17::optional<float> lambda_expressions = cpp17::optional<float>(),
248  cpp17::optional<int> num_expression_coefficients_to_fit = cpp17::optional<int>())
249 {
250  std::vector<float> expression_coefficients;
251  if (cpp17::holds_alternative<morphablemodel::PcaModel>(expression_model))
252  {
253  const auto& pca_expression_model = cpp17::get<morphablemodel::PcaModel>(expression_model);
254  // Usually, one would pass the result of the PCA identity shape fitting as a face_instance to this
255  // function. We then need to add the mean of the expression PCA model before performing the fitting,
256  // since we solve for the differences.
257  const Eigen::VectorXf face_instance_with_expression_mean =
258  face_instance + pca_expression_model.get_mean();
259  // Todo: Add lambda, num_coeffs_to_fit, ...
260  return fit_shape_to_landmarks_linear(pca_expression_model, affine_camera_matrix, landmarks,
261  vertex_ids, face_instance_with_expression_mean,
262  lambda_expressions.value_or(65.0f),
263  num_expression_coefficients_to_fit);
264  } else if (cpp17::holds_alternative<morphablemodel::Blendshapes>(expression_model))
265  {
266  const auto& expression_blendshapes = cpp17::get<morphablemodel::Blendshapes>(expression_model);
267  return fit_blendshapes_to_landmarks_nnls(expression_blendshapes, face_instance, affine_camera_matrix,
268  landmarks, vertex_ids);
269  } else
270  {
271  throw std::runtime_error("The given expression_model doesn't contain a PcaModel or Blendshapes.");
272  }
273 };

322 inline std::pair<core::Mesh, fitting::RenderingParameters> fit_shape_and_pose(
323  const morphablemodel::MorphableModel& morphable_model,
324  const core::LandmarkCollection<Eigen::Vector2f>& landmarks, const core::LandmarkMapper& landmark_mapper,
325  int image_width, int image_height, const morphablemodel::EdgeTopology& edge_topology,
326  const fitting::ContourLandmarks& contour_landmarks, const fitting::ModelContour& model_contour,
327  int num_iterations, cpp17::optional<int> num_shape_coefficients_to_fit, float lambda_identity,
328  cpp17::optional<int> num_expression_coefficients_to_fit, cpp17::optional<float> lambda_expressions,
329  std::vector<float>& pca_shape_coefficients, std::vector<float>& expression_coefficients,
330  std::vector<Eigen::Vector2f>& fitted_image_points)
331 {
332  // assert(blendshapes.size() > 0);
333  assert(landmarks.size() >= 4);
334  assert(image_width > 0 && image_height > 0);
335  assert(num_iterations > 0); // Can we allow 0, for only the initial pose-fit?
336  assert(pca_shape_coefficients.size() <= morphable_model.get_shape_model().get_num_principal_components());
337  // More asserts I forgot?
338  if (num_shape_coefficients_to_fit)
339  {
340  if (num_shape_coefficients_to_fit.value() >
341  morphable_model.get_shape_model().get_num_principal_components())
342  {
343  throw std::runtime_error(
344  "Specified more shape coefficients to fit than the given shape model contains.");
345  }
346  }
347 
348  using Eigen::MatrixXf;
349  using Eigen::Vector2f;
350  using Eigen::Vector4f;
351  using Eigen::VectorXf;
352  using std::vector;
353 
354  if (!num_shape_coefficients_to_fit)
355  {
356  num_shape_coefficients_to_fit = morphable_model.get_shape_model().get_num_principal_components();
357  }
358 
359  if (pca_shape_coefficients.empty())
360  {
361  pca_shape_coefficients.resize(num_shape_coefficients_to_fit.value());
362  }
363  // Todo: This leaves the following case open: num_coeffs given is empty or defined, but the
364  // pca_shape_coefficients given is != num_coeffs or the model's max-coeffs. What to do then? Handle & document!
365 
366  /*if (expression_coefficients.empty())
367  {
368  expression_coefficients.resize(blendshapes.size());
369  }*/
370 
371  // Current mesh - either from the given coefficients, or the mean:
372  VectorXf current_pca_shape = morphable_model.get_shape_model().draw_sample(pca_shape_coefficients);
373  assert(morphable_model.has_separate_expression_model()); // Note: We could also just skip the expression fitting in this case.
374  // Note we don't check whether the shape and expression model dimensions match.
375  // Note: We're calling this in a loop, and morphablemodel::to_matrix(expression_blendshapes) now gets
376  // called again in every fitting iteration.
377  VectorXf current_combined_shape =
378  current_pca_shape +
379  draw_sample(morphable_model.get_expression_model().value(), expression_coefficients);
380  auto current_mesh = morphablemodel::sample_to_mesh(
381  current_combined_shape, morphable_model.get_color_model().get_mean(),
382  morphable_model.get_shape_model().get_triangle_list(),
383  morphable_model.get_color_model().get_triangle_list(), morphable_model.get_texture_coordinates(),
384  morphable_model.get_texture_triangle_indices());
385 
386  // The 2D and 3D point correspondences used for the fitting:
387  vector<Vector4f> model_points; // the points in the 3D shape model
388  vector<int> vertex_indices; // their vertex indices
389  vector<Vector2f> image_points; // the corresponding 2D landmark points
390 
391  // Sub-select all the landmarks which we have a mapping for (i.e. that are defined in the 3DMM),
392  // and get the corresponding model points (mean if given no initial coeffs, from the computed shape otherwise):
393  for (int i = 0; i < landmarks.size(); ++i)
394  {
395  auto converted_name = landmark_mapper.convert(landmarks[i].name);
396  if (!converted_name)
397  { // no mapping defined for the current landmark
398  continue;
399  }
400  // If the MorphableModel does not contain landmark definitions, we expect the user to have given us
401  // direct mappings (e.g. directly from ibug identifiers to vertex ids). If the model does contain
402  // landmark definitions, we expect the user to use mappings from their landmark identifiers (e.g.
403  // ibug) to the landmark definitions, and not to vertex indices.
404  // Todo: This might be worth mentioning in the function documentation of fit_shape_and_pose.
405  int vertex_idx;
406  if (morphable_model.get_landmark_definitions())
407  {
408  const auto found_vertex_idx =
409  morphable_model.get_landmark_definitions().value().find(converted_name.value());
410  if (found_vertex_idx != std::end(morphable_model.get_landmark_definitions().value()))
411  {
412  vertex_idx = found_vertex_idx->second;
413  } else
414  {
415  continue;
416  }
417  } else
418  {
419  vertex_idx = std::stoi(converted_name.value());
420  }
421  Vector4f vertex(current_mesh.vertices[vertex_idx][0], current_mesh.vertices[vertex_idx][1],
422  current_mesh.vertices[vertex_idx][2], 1.0f);
423  model_points.emplace_back(vertex);
424  vertex_indices.emplace_back(vertex_idx);
425  image_points.emplace_back(landmarks[i].coordinates);
426  }
427 
428  // Need to do an initial pose fit to do the contour fitting inside the loop.
429  // We'll do an expression fit too, since face shapes vary quite a lot, depending on expressions.
430  fitting::ScaledOrthoProjectionParameters current_pose =
431  fitting::estimate_orthographic_projection_linear(image_points, model_points, true, image_height);
432  fitting::RenderingParameters rendering_params(current_pose, image_width, image_height);
433 
434  const Eigen::Matrix<float, 3, 4> affine_from_ortho =
435  fitting::get_3x4_affine_camera_matrix(rendering_params, image_width, image_height);
436  expression_coefficients =
437  fit_expressions(morphable_model.get_expression_model().value(), current_pca_shape, affine_from_ortho,
438  image_points, vertex_indices, lambda_expressions, num_expression_coefficients_to_fit);
439 
440  // Mesh with same PCA coeffs as before, but new expression fit (this is relevant if no initial blendshape coeffs have been given):
441  current_combined_shape = current_pca_shape + draw_sample(morphable_model.get_expression_model().value(),
442  expression_coefficients);
443  current_mesh = morphablemodel::sample_to_mesh(
444  current_combined_shape, morphable_model.get_color_model().get_mean(),
445  morphable_model.get_shape_model().get_triangle_list(),
446  morphable_model.get_color_model().get_triangle_list(), morphable_model.get_texture_coordinates(),
447  morphable_model.get_texture_triangle_indices());
448 
449  // The static (fixed) landmark correspondences which will stay the same throughout
450  // the fitting (the inner face landmarks):
451  const auto fixed_image_points = image_points;
452  const auto fixed_vertex_indices = vertex_indices;
453 
454  for (int i = 0; i < num_iterations; ++i)
455  {
456  image_points = fixed_image_points;
457  vertex_indices = fixed_vertex_indices;
458  // Given the current pose, find 2D-3D contour correspondences of the front-facing face contour:
459  vector<Vector2f> image_points_contour;
460  vector<int> vertex_indices_contour;
461  const auto yaw_angle = glm::degrees(glm::eulerAngles(rendering_params.get_rotation())[1]);
462  // For each 2D contour landmark, get the corresponding 3D vertex point and vertex id:
463  std::tie(image_points_contour, std::ignore, vertex_indices_contour) =
464  fitting::get_contour_correspondences(landmarks, contour_landmarks, model_contour, yaw_angle,
465  current_mesh, rendering_params.get_modelview(),
466  rendering_params.get_projection(),
467  fitting::get_opencv_viewport(image_width, image_height));
468  // Add the contour correspondences to the set of landmarks that we use for the fitting:
469  vertex_indices = fitting::concat(vertex_indices, vertex_indices_contour);
470  image_points = fitting::concat(image_points, image_points_contour);
471 
472  // Fit the occluding (away-facing) contour using the detected contour LMs:
473  vector<Vector2f> occluding_contour_landmarks;
474  if (yaw_angle >= 0.0f) // positive yaw = subject looking to the left
475  { // the left contour is the occluding one we want to use ("away-facing")
476  const auto contour_landmarks_ =
477  core::filter(landmarks, contour_landmarks.left_contour); // Can do this outside of the loop
478  std::for_each(begin(contour_landmarks_), end(contour_landmarks_),
479  [&occluding_contour_landmarks](const auto& lm) {
480  occluding_contour_landmarks.push_back({lm.coordinates[0], lm.coordinates[1]});
481  });
482  } else
483  {
484  const auto contour_landmarks_ = core::filter(landmarks, contour_landmarks.right_contour);
485  std::for_each(begin(contour_landmarks_), end(contour_landmarks_),
486  [&occluding_contour_landmarks](const auto& lm) {
487  occluding_contour_landmarks.push_back({lm.coordinates[0], lm.coordinates[1]});
488  });
489  }
490  const auto edge_correspondences = fitting::find_occluding_edge_correspondences(
491  current_mesh, edge_topology, rendering_params, occluding_contour_landmarks, 180.0f);
492  image_points = fitting::concat(image_points, edge_correspondences.first);
493  vertex_indices = fitting::concat(vertex_indices, edge_correspondences.second);
494 
495  // Get the model points of the current mesh, for all correspondences that we've got:
496  model_points.clear();
497  for (auto v : vertex_indices)
498  {
499  model_points.push_back({current_mesh.vertices[v][0], current_mesh.vertices[v][1],
500  current_mesh.vertices[v][2], 1.0f});
501  }
502 
503  // Re-estimate the pose, using all correspondences:
504  current_pose =
505  fitting::estimate_orthographic_projection_linear(image_points, model_points, true, image_height);
506  rendering_params = fitting::RenderingParameters(current_pose, image_width, image_height);
507 
508  const Eigen::Matrix<float, 3, 4> affine_from_ortho =
509  fitting::get_3x4_affine_camera_matrix(rendering_params, image_width, image_height);
510 
511  // Estimate the PCA shape coefficients with the current blendshape coefficients:
512  const VectorXf mean_plus_expressions =
513  morphable_model.get_shape_model().get_mean() +
514  draw_sample(morphable_model.get_expression_model().value(), expression_coefficients);
515  pca_shape_coefficients = fitting::fit_shape_to_landmarks_linear(
516  morphable_model.get_shape_model(), affine_from_ortho, image_points, vertex_indices,
517  mean_plus_expressions, lambda_identity, num_shape_coefficients_to_fit);
518 
519  // Estimate the blendshape coefficients with the current PCA model estimate:
520  current_pca_shape = morphable_model.get_shape_model().draw_sample(pca_shape_coefficients);
521  expression_coefficients = fit_expressions(
522  morphable_model.get_expression_model().value(), current_pca_shape, affine_from_ortho,
523  image_points, vertex_indices, lambda_expressions, num_expression_coefficients_to_fit);
524 
525  current_combined_shape =
526  current_pca_shape +
527  draw_sample(morphable_model.get_expression_model().value(), expression_coefficients);
528  current_mesh = morphablemodel::sample_to_mesh(
529  current_combined_shape, morphable_model.get_color_model().get_mean(),
530  morphable_model.get_shape_model().get_triangle_list(),
531  morphable_model.get_color_model().get_triangle_list(), morphable_model.get_texture_coordinates(),
532  morphable_model.get_texture_triangle_indices());
533  }
534 
535  fitted_image_points = image_points;
536  return {current_mesh, rendering_params}; // I think we could also work with a VectorXf face_instance in
537  // this function instead of a Mesh, but it would convolute the
538  // code more (i.e. more complicated to access vertices).
539 };

667  // Todo: This leaves the following case open: num_coeffs given is empty or defined, but the

674 
675  // Current mesh - either from the given coefficients, or the mean:
676  VectorXf current_pca_shape = morphable_model.get_shape_model().draw_sample(pca_shape_coefficients);
677  assert(morphable_model.has_separate_expression_model()); // Note: We could also just skip the expression fitting in this case.
678  // Note we don't check whether the shape and expression model dimensions match.
679  // Note: We're calling this in a loop, and morphablemodel::to_matrix(expression_blendshapes) now gets
680  // called again in every fitting iteration.
681  VectorXf current_combined_shape =
682  current_pca_shape +
683  draw_sample(morphable_model.get_expression_model().value(), expression_coefficients);
684  auto current_mesh = morphablemodel::sample_to_mesh(
685  current_combined_shape, morphable_model.get_color_model().get_mean(),
686  morphable_model.get_shape_model().get_triangle_list(),
687  morphable_model.get_color_model().get_triangle_list(), morphable_model.get_texture_coordinates(),
688  morphable_model.get_texture_triangle_indices());
689 
690  // The 2D and 3D point correspondences used for the fitting:
691  vector<Vector4f> model_points; // the points in the 3D shape model
692  // Get the model points corresponding to the given image points (mean if given no initial coeffs, from the computed shape otherwise):
693  for (int i = 0; i < image_points.size(); ++i)
694  {
695  const int vertex_idx = vertex_indices[i];
696  Vector4f vertex(current_mesh.vertices[vertex_idx][0], current_mesh.vertices[vertex_idx][1],
697  current_mesh.vertices[vertex_idx][2], 1.0f);
698  model_points.emplace_back(vertex);
699  }
700 
701  // Need to do an initial pose fit to do the contour fitting inside the loop.
702  // We'll do an expression fit too, since face shapes vary quite a lot, depending on expressions.
703  fitting::ScaledOrthoProjectionParameters current_pose =
704  fitting::estimate_orthographic_projection_linear(image_points, model_points, true, image_height);
705  fitting::RenderingParameters rendering_params(current_pose, image_width, image_height);
706 
707  const Eigen::Matrix<float, 3, 4> affine_from_ortho =
708  fitting::get_3x4_affine_camera_matrix(rendering_params, image_width, image_height);
709  expression_coefficients =
710  fit_expressions(morphable_model.get_expression_model().value(), current_pca_shape, affine_from_ortho,
711  image_points, vertex_indices, lambda_expressions, num_expression_coefficients_to_fit);
712 
713  // Mesh with same PCA coeffs as before, but new expression fit (this is relevant if no initial blendshape coeffs have been given):
714  current_combined_shape = current_pca_shape + draw_sample(morphable_model.get_expression_model().value(),
715  expression_coefficients);
716  current_mesh = morphablemodel::sample_to_mesh(
717  current_combined_shape, morphable_model.get_color_model().get_mean(),
718  morphable_model.get_shape_model().get_triangle_list(),
719  morphable_model.get_color_model().get_triangle_list(), morphable_model.get_texture_coordinates(),
720  morphable_model.get_texture_triangle_indices());
721 
722  for (int i = 0; i < num_iterations; ++i)
723  {
724  // Get the model points of the current mesh, for all correspondences that we've got:
725  model_points.clear();
726  for (auto v : vertex_indices)
727  {
728  model_points.push_back({current_mesh.vertices[v][0], current_mesh.vertices[v][1],
729  current_mesh.vertices[v][2], 1.0f});
730  }
731 
732  // Re-estimate the pose, using all correspondences:
733  current_pose =
734  fitting::estimate_orthographic_projection_linear(image_points, model_points, true, image_height);
735  rendering_params = fitting::RenderingParameters(current_pose, image_width, image_height);
736 
737  const Eigen::Matrix<float, 3, 4> affine_from_ortho =
738  fitting::get_3x4_affine_camera_matrix(rendering_params, image_width, image_height);
739 
740  // Estimate the PCA shape coefficients with the current blendshape coefficients:
741  const VectorXf mean_plus_expressions =
742  morphable_model.get_shape_model().get_mean() +
743  draw_sample(morphable_model.get_expression_model().value(), expression_coefficients);
744  pca_shape_coefficients = fitting::fit_shape_to_landmarks_linear(
745  morphable_model.get_shape_model(), affine_from_ortho, image_points, vertex_indices,
746  mean_plus_expressions, lambda_identity, num_shape_coefficients_to_fit);
747 
748  // Estimate the blendshape coefficients with the current PCA model estimate:
749  current_pca_shape = morphable_model.get_shape_model().draw_sample(pca_shape_coefficients);
750  expression_coefficients = fit_expressions(
751  morphable_model.get_expression_model().value(), current_pca_shape, affine_from_ortho,
752  image_points, vertex_indices, lambda_expressions, num_expression_coefficients_to_fit);
753 
754  current_combined_shape =
755  current_pca_shape +
756  draw_sample(morphable_model.get_expression_model().value(), expression_coefficients);
757  current_mesh = morphablemodel::sample_to_mesh(
758  current_combined_shape, morphable_model.get_color_model().get_mean(),
759  morphable_model.get_shape_model().get_triangle_list(),
760  morphable_model.get_color_model().get_triangle_list(), morphable_model.get_texture_coordinates(),
761  morphable_model.get_texture_triangle_indices());
762  }
763 
764  fitted_image_points = image_points;
765  return {current_mesh, rendering_params}; // I think we could also work with a VectorXf face_instance in
766  // this function instead of a Mesh, but it would convolute the
767  // code more (i.e. more complicated to access vertices).
768 };
