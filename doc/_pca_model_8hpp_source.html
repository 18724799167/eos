369 inline Eigen::MatrixXf rescale_pca_basis(const Eigen::MatrixXf& orthonormal_basis,
370  const Eigen::VectorXf& eigenvalues)
371 {
372  Eigen::MatrixXf rescaled_basis(orthonormal_basis.rows(),
373  orthonormal_basis.cols()); // empty matrix with the same dimensions
374  const Eigen::VectorXf sqrt_of_eigenvalues = eigenvalues.array().sqrt();
375  // Rescale the basis: We multiply each eigenvector (i.e. each column) with the square root of its
376  // corresponding eigenvalue
377  for (int basis = 0; basis < orthonormal_basis.cols(); ++basis)
378  {
379  rescaled_basis.col(basis) = orthonormal_basis.col(basis) * sqrt_of_eigenvalues(basis);
380  }
381 
382  return rescaled_basis;
383 };
384 
394 inline Eigen::MatrixXf normalise_pca_basis(const Eigen::MatrixXf& rescaled_basis,
395  const Eigen::VectorXf& eigenvalues)
396 {
397  Eigen::MatrixXf orthonormal_basis(rescaled_basis.rows(),
398  rescaled_basis.cols()); // empty matrix with the same dimensions
399  const Eigen::VectorXf one_over_sqrt_of_eigenvalues = eigenvalues.array().rsqrt();
400  // "Normalise" the basis: We multiply each eigenvector (i.e. each column) with 1 over the square root of
401  // its corresponding eigenvalue
402  for (int basis = 0; basis < rescaled_basis.cols(); ++basis)
403  {
404  orthonormal_basis.col(basis) = rescaled_basis.col(basis) * one_over_sqrt_of_eigenvalues(basis);
405  }
406 
407  return orthonormal_basis;
408 };
