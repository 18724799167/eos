53 inline cv::Mat estimate_affine_camera(std::vector<cv::Vec2f> image_points,
54  std::vector<cv::Vec4f> model_points)
55 {
56  using cv::Mat;
57  assert(image_points.size() == model_points.size());
58  assert(image_points.size() >= 4); // Number of correspondence points given needs to be equal to or larger than 4
59 
60  const int num_correspondences = static_cast<int>(image_points.size());
61 
62  Mat matImagePoints; // will be numCorrespondences x 2, CV_32FC1
63  Mat matModelPoints; // will be numCorrespondences x 3, CV_32FC1
64 
65  for (int i = 0; i < image_points.size(); ++i)
66  {
67  Mat imgPoint(1, 2, CV_32FC1);
68  imgPoint.at<float>(0, 0) = image_points[i][0];
69  imgPoint.at<float>(0, 1) = image_points[i][1];
70  matImagePoints.push_back(imgPoint);
71 
72  Mat mdlPoint(1, 3, CV_32FC1);
73  mdlPoint.at<float>(0, 0) = model_points[i][0];
74  mdlPoint.at<float>(0, 1) = model_points[i][1];
75  mdlPoint.at<float>(0, 2) = model_points[i][2];
76  matModelPoints.push_back(mdlPoint);
77  }
78 
79  // translate the centroid of the image points to the origin:
80  Mat imagePointsMean; // use non-homogeneous coords for the next few steps? (less submatrices etc overhead)
81  cv::reduce(matImagePoints, imagePointsMean, 0, CV_REDUCE_AVG);
82  imagePointsMean = cv::repeat(imagePointsMean, matImagePoints.rows, 1); // get T_13 and T_23 from imagePointsMean
83  matImagePoints = matImagePoints - imagePointsMean;
84  // scale the image points such that the RMS distance from the origin is sqrt(2):
85  // 1) calculate the average norm (root mean squared distance) of all vectors
86  float average_norm = 0.0f;
87  for (int row = 0; row < matImagePoints.rows; ++row)
88  {
89  average_norm += static_cast<float>(cv::norm(matImagePoints.row(row), cv::NORM_L2));
90  }
91  average_norm /= matImagePoints.rows;
92  // 2) multiply every vectors coordinate by sqrt(2)/average_norm
93  float scaleFactor = static_cast<float>(std::sqrt(2)) / average_norm;
94  matImagePoints *= scaleFactor; // add unit homogeneous component here
95  // The points in matImagePoints now have a RMS distance from the origin of sqrt(2).
96  // The normalisation matrix so that the 2D points are mean-free and their norm is as described above.
97  Mat T = Mat::zeros(3, 3, CV_32FC1);
98  T.at<float>(0, 0) = scaleFactor; // s_x
99  T.at<float>(1, 1) = scaleFactor; // s_y
100  T.at<float>(0, 2) = -imagePointsMean.at<float>(0, 0) * scaleFactor; // t_x
101  T.at<float>(1, 2) = -imagePointsMean.at<float>(0, 1) * scaleFactor; // t_y
102  T.at<float>(2, 2) = 1;
103 
104  // center the model points to the origin:
105  Mat tmpOrigMdlPoints = matModelPoints.clone(); // Temp for testing: Save the original coordinates.
106  // translate the centroid of the model points to the origin:
107  Mat modelPointsMean; // use non-homogeneous coords for the next few steps? (less submatrices etc overhead)
108  cv::reduce(matModelPoints, modelPointsMean, 0, CV_REDUCE_AVG);
109  modelPointsMean = cv::repeat(modelPointsMean, matModelPoints.rows, 1);
110  matModelPoints = matModelPoints - modelPointsMean;
111  // scale the model points such that the RMS distance from the origin is sqrt(3):
112  // 1) calculate the average norm (root mean squared distance) of all vectors
113  average_norm = 0.0f;
114  for (int row = 0; row < matModelPoints.rows; ++row)
115  {
116  average_norm += static_cast<float>(cv::norm(matModelPoints.row(row), cv::NORM_L2));
117  }
118  average_norm /= matModelPoints.rows;
119  // 2) multiply every vectors coordinate by sqrt(3)/avgnorm
120  scaleFactor = static_cast<float>(std::sqrt(3)) / average_norm;
121  matModelPoints *= scaleFactor; // add unit homogeneous component here
122  // The points in matModelPoints now have a RMS distance from the origin of sqrt(3).
123  // The normalisation matrix so that the 3D points are mean-free and their norm is as described above.
124  Mat U = Mat::zeros(4, 4, CV_32FC1);
125  U.at<float>(0, 0) = scaleFactor; // s_x
126  U.at<float>(1, 1) = scaleFactor; // s_y
127  U.at<float>(2, 2) = scaleFactor; // s_z
128  U.at<float>(0, 3) = -modelPointsMean.at<float>(0, 0) * scaleFactor; // t_x
129  U.at<float>(1, 3) = -modelPointsMean.at<float>(0, 1) * scaleFactor; // t_y
130  U.at<float>(2, 3) = -modelPointsMean.at<float>(0, 2) * scaleFactor; // t_z
131  U.at<float>(3, 3) = 1;
132 
133  // Estimate the normalised camera matrix (C tilde).
134  // We are solving the system $A_8 * p_8 = b$
135  // The solution is obtained by the pseudo-inverse of A_8:
136  // $p_8 = A_8^+ * b$
137  Mat A_8 = Mat::zeros(num_correspondences * 2, 8, CV_32FC1);
138  // Mat p_8(); // p_8 is 8 x 1. We are solving for it.
139  Mat b(num_correspondences * 2, 1, CV_32FC1);
140  for (int i = 0; i < num_correspondences; ++i)
141  {
142  A_8.at<float>(2 * i, 0) = matModelPoints.at<float>(i, 0); // could maybe made faster by assigning the whole row/col-range if possible?
143  A_8.at<float>(2 * i, 1) = matModelPoints.at<float>(i, 1);
144  A_8.at<float>(2 * i, 2) = matModelPoints.at<float>(i, 2);
145  A_8.at<float>(2 * i, 3) = 1;
146  A_8.at<float>((2 * i) + 1, 4) = matModelPoints.at<float>(i, 0);
147  A_8.at<float>((2 * i) + 1, 5) = matModelPoints.at<float>(i, 1);
148  A_8.at<float>((2 * i) + 1, 6) = matModelPoints.at<float>(i, 2);
149  A_8.at<float>((2 * i) + 1, 7) = 1;
150  b.at<float>(2 * i, 0) = matImagePoints.at<float>(i, 0);
151  b.at<float>((2 * i) + 1, 0) = matImagePoints.at<float>(i, 1);
152  }
153  Mat p_8 = A_8.inv(cv::DECOMP_SVD) * b;
154  Mat C_tilde = Mat::zeros(3, 4, CV_32FC1);
155  C_tilde.row(0) = p_8.rowRange(0, 4).t(); // The first row of C_tilde consists of the first 4 entries of the col-vector p_8
156  C_tilde.row(1) = p_8.rowRange(4, 8).t(); // Second row = last 4 entries
157  C_tilde.at<float>(2, 3) = 1; // the last row is [0 0 0 1]
158 
159  Mat P_Affine = T.inv() * C_tilde * U;
160  return P_Affine;
161 };
