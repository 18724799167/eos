130  vector<Triangle<T, P>> triangles_to_raster;
131  // This builds the (one and final) triangles to render. Meaning: The triangles formed of mesh.tvi (the
132  // ones that survived the clip/culling), plus possibly more that intersect one of the frustum planes
133  // (i.e. this can generate new triangles with new pos/vc/texcoords).
134  for (const auto& tri_indices : mesh.tvi)
135  {
136  unsigned char visibility_bits[3];
137  for (unsigned char k = 0; k < 3; k++)
138  {
139  visibility_bits[k] = 0;
140  const auto x_cc = clipspace_vertices[tri_indices[k]].x;
141  const auto y_cc = clipspace_vertices[tri_indices[k]].y;
142  const auto z_cc = clipspace_vertices[tri_indices[k]].z;
143  const auto w_cc = clipspace_vertices[tri_indices[k]].w;
144  if (x_cc < -w_cc) // true if outside of view frustum. False if on or inside the plane.
145  visibility_bits[k] |= 1; // set bit if outside of frustum
146  if (x_cc > w_cc)
147  visibility_bits[k] |= 2;
148  if (y_cc < -w_cc)
149  visibility_bits[k] |= 4;
150  if (y_cc > w_cc)
151  visibility_bits[k] |= 8;
152  if (enable_near_clipping && z_cc < -w_cc) // near plane frustum clipping
153  visibility_bits[k] |= 16;
154  if (rasterizer.enable_far_clipping && z_cc > w_cc) // far plane frustum clipping
155  visibility_bits[k] |= 32;
156  } // if all bits are 0, then it's inside the frustum
157  // all vertices are not visible - reject the triangle.
158  if ((visibility_bits[0] & visibility_bits[1] & visibility_bits[2]) > 0)
159  {
160  continue;
161  }
162  // all vertices are visible - pass the whole triangle to the rasteriser. = All bits of all 3
163  // triangles are 0.
164  if ((visibility_bits[0] | visibility_bits[1] | visibility_bits[2]) == 0)
165  {

167  std::array<glm::tvec4<T, P>, 3> prospective_tri{
168  divide_by_w(clipspace_vertices[tri_indices[0]]),
169  divide_by_w(clipspace_vertices[tri_indices[1]]),
170  divide_by_w(clipspace_vertices[tri_indices[2]])};
171  // We have a prospective tri in NDC coords now, with its vertices having coords [x_ndc, y_ndc,
172  // z_ndc, 1/w_clip].
173 
174  // Replaces x and y of the NDC coords with the screen coords. Keep z and w the same.

176  clip_to_screen_space(prospective_tri[0].x, prospective_tri[0].y,
177  rasterizer.viewport_width, rasterizer.viewport_height);

179  prospective_tri[0].y = v0_screen.y;

182  rasterizer.viewport_width, rasterizer.viewport_height);
183  prospective_tri[1].x = v1_screen.x;
184  prospective_tri[1].y = v1_screen.y;

187  rasterizer.viewport_width, rasterizer.viewport_height);

189  prospective_tri[2].y = v2_screen.y;

192  // elsewhere? No?
193  if (enable_backface_culling)
194  {

204  glm::tvec2<T, P>(prospective_tri[0].x, prospective_tri[0].y),

207  rasterizer.viewport_height);

209  const auto max_x = boundingBox.x + boundingBox.width;
210  const auto min_y = boundingBox.y;
211  const auto max_y = boundingBox.y + boundingBox.height;
212  if (max_x <= min_x || max_y <= min_y)
213  { // Note: Can the width/height of the bbox be negative? Maybe we only need to check for
214  // equality here?
215  continue;
216  }

219  triangles_to_raster.push_back(Triangle<T, P>{
220  detail::Vertex<T, P>{prospective_tri[0],
221  glm::tvec3<T, P>(mesh.colors[tri_indices[0]](0),
222  mesh.colors[tri_indices[0]](1),

224  glm::tvec2<T, P>(mesh.texcoords[tri_indices[0]](0),
225  mesh.texcoords[tri_indices[0]](1))},
226  detail::Vertex<T, P>{prospective_tri[1],

253  glm::tvec3<T, P>(mesh.colors[tri_indices[1]](0), mesh.colors[tri_indices[1]](1),

268  // little bit unsure because Songho says the normal
269  // of the near-plane is (0,0,-1,1) (maybe I have to

370 template <typename T, glm::precision P = glm::defaultp>
371 std::vector<detail::Vertex<T, P>>
372 clip_polygon_to_plane_in_4d(const std::vector<detail::Vertex<T, P>>& vertices,
373  const glm::tvec4<T, P>& plane_normal)
374 {
375  std::vector<detail::Vertex<T, P>> clipped_vertices;
376 
377  // We can have 2 cases:
378  // * 1 vertex visible: we make 1 new triangle out of the visible vertex plus the 2 intersection points
379  // with the near-plane
380  // * 2 vertices visible: we have a quad, so we have to make 2 new triangles out of it.
381 
382  // See here for more info?
383  // http://math.stackexchange.com/questions/400268/equation-for-a-line-through-a-plane-in-homogeneous-coordinates
384 
385  for (unsigned int i = 0; i < vertices.size(); i++)
386  {
387  int a = i; // the current vertex
388  int b = (i + 1) % vertices.size(); // the following vertex (wraps around 0)
389 
390  using glm::dot;
391  const T fa = dot(vertices[a].position, plane_normal); // Note: Shouldn't they be unit length?
392  const T fb = dot(vertices[b].position,
393  plane_normal); // < 0 means on visible side, > 0 means on invisible side?
394 
395  if ((fa < 0 && fb > 0) || (fa > 0 && fb < 0)) // one vertex is on the visible side of the plane, one
396  // on the invisible? so we need to split?
397  {
398  const glm::tvec4<T, P> direction = vertices[b].position - vertices[a].position;
399  const T t = -(dot(plane_normal, vertices[a].position)) /
400  (dot(plane_normal, direction)); // the parametric value on the line, where the line to
401  // draw intersects the plane?
402 
403  // generate a new vertex at the line-plane intersection point
404  const glm::tvec4<T, P> position = vertices[a].position + t * direction;
405  const glm::tvec3<T, P> color = vertices[a].color + t * (vertices[b].color - vertices[a].color);
406  const glm::tvec2<T, P> texcoords =
407  vertices[a].texcoords +
408  t * (vertices[b].texcoords -
409  vertices[a].texcoords); // We could omit that if we don't render with texture.
410 
411  if (fa < 0) // we keep the original vertex plus the new one
412  {
413  clipped_vertices.push_back(vertices[a]);
414  clipped_vertices.push_back(detail::Vertex<T, P>{position, color, texcoords});
415  } else if (fb < 0) // we use only the new vertex
416  {
417  clipped_vertices.push_back(detail::Vertex<T, P>{position, color, texcoords});
418  }
419  } else if (fa < 0 && fb < 0) // both are visible (on the "good" side of the plane), no splitting
420  // required, use the current vertex
421  {
422  clipped_vertices.push_back(vertices[a]);
423  }
424  // else, both vertices are not visible, nothing to add and draw
425  }
426 
427  return clipped_vertices;
428 };
