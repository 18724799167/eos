123 inline std::pair<core::Image4u, core::Image1d>
124 render(core::Mesh mesh, glm::tmat4x4<float> model_view_matrix, glm::tmat4x4<float> projection_matrix,
125  int viewport_width, int viewport_height, const cpp17::optional<Texture>& texture = cpp17::nullopt,
126  bool enable_backface_culling = false, bool enable_near_clipping = true,
127  bool enable_far_clipping = true)
128 {
129  // Some internal documentation / old todos or notes:
130  // - maybe change and pass depthBuffer as an optional arg (&?), because usually we never need it outside
131  // the renderer. Or maybe even a getDepthBuffer().
132  // - modelViewMatrix goes to eye-space (camera space), projection does ortho or perspective proj.
133  // - bool enable_texturing = false; Maybe re-add later, not sure
134  // - take a cv::Mat texture instead and convert to Texture internally? no, we don't want to recreate
135  // mipmap levels on each render() call.
136 
137  assert(mesh.vertices.size() == mesh.colors.size() ||
138  mesh.colors.empty()); // The number of vertices has to be equal for both shape and colour, or,
139  // alternatively, it has to be a shape-only model.
140  assert(mesh.vertices.size() == mesh.texcoords.size() || mesh.texcoords.empty()); // same for the texcoords
141  // another assert: If cv::Mat texture != empty, then we need texcoords?
142 
143  using std::vector;
144 
145  core::Image4u colorbuffer =
146  core::image::zeros<core::Pixel<std::uint8_t, 4>>(viewport_height, viewport_width);
147  core::Image1d depthbuffer =
148  core::image::constant(viewport_height, viewport_width, std::numeric_limits<double>::max());
149 
150  // Vertex shader:
151  // processedVertex = shade(Vertex); // processedVertex : pos, col, tex, texweight
152  // Assemble the vertices, project to clip space, and store as detail::Vertex (the internal
153  // representation):
154  vector<detail::Vertex<float>> clipspace_vertices;
155  clipspace_vertices.reserve(mesh.vertices.size());
156  for (int i = 0; i < mesh.vertices.size(); ++i)
157  { // "previously": mesh.vertex
158  const glm::tvec4<float> vertex(mesh.vertices[i][0], mesh.vertices[i][1], mesh.vertices[i][2], 1.0f);
159  const glm::tvec4<float> clipspace_coords = projection_matrix * model_view_matrix * vertex;
160  glm::tvec3<float> vertex_colour;
161  if (mesh.colors.empty())
162  {
163  vertex_colour = glm::tvec3<float>(0.5f, 0.5f, 0.5f);
164  } else
165  {
166  vertex_colour = glm::tvec3<float>(mesh.colors[i][0], mesh.colors[i][1], mesh.colors[i][2]);
167  }
168  clipspace_vertices.push_back(detail::Vertex<float>{
169  clipspace_coords, vertex_colour, glm::tvec2<float>(mesh.texcoords[i][0], mesh.texcoords[i][1])});
170  }
171 
172  // All vertices are in clip-space now.
173  // Prepare the rasterisation stage.
174  // For every vertex/tri:
175  vector<detail::TriangleToRasterize> triangles_to_raster;
176  for (const auto& tri_indices : mesh.tvi)
177  {
178  // Todo: Split this whole stuff up. Make a "clip" function, ... rename "processProspective..".. what
179  // is "process"... get rid of "continue;"-stuff by moving stuff inside process...
180  // classify vertices visibility with respect to the planes of the view frustum
181  // we're in clip-coords (NDC), so just check if outside [-1, 1] x ...
182  // Actually we're in clip-coords and it's not the same as NDC. We're only in NDC after the division by w.
183  // We should do the clipping in clip-coords though. See
184  // http://www.songho.ca/opengl/gl_projectionmatrix.html for more details.
185  // However, when comparing against w_c below, we might run into the trouble of the sign again in the
186  // affine case.
187 
188  // 'w' is always positive, as it is -z_camspace, and all z_camspace are negative.
189  unsigned char visibility_bits[3];
190  for (unsigned char k = 0; k < 3; k++)
191  {
192  visibility_bits[k] = 0;
193  const float x_cc = clipspace_vertices[tri_indices[k]].position[0];
194  const float y_cc = clipspace_vertices[tri_indices[k]].position[1];
195  const float z_cc = clipspace_vertices[tri_indices[k]].position[2];
196  const float w_cc = clipspace_vertices[tri_indices[k]].position[3];
197  if (x_cc < -w_cc) // true if outside of view frustum. False if on or inside the plane.
198  visibility_bits[k] |= 1; // set bit if outside of frustum
199  if (x_cc > w_cc)
200  visibility_bits[k] |= 2;
201  if (y_cc < -w_cc)
202  visibility_bits[k] |= 4;
203  if (y_cc > w_cc)
204  visibility_bits[k] |= 8;
205  if (enable_near_clipping && z_cc < -w_cc) // near plane frustum clipping
206  visibility_bits[k] |= 16;
207  if (enable_far_clipping && z_cc > w_cc) // far plane frustum clipping
208  visibility_bits[k] |= 32;
209  } // if all bits are 0, then it's inside the frustum
210  // all vertices are not visible - reject the triangle.
211  if ((visibility_bits[0] & visibility_bits[1] & visibility_bits[2]) > 0)
212  {
213  continue;
214  }
215  // all vertices are visible - pass the whole triangle to the rasterizer. = All bits of all 3 triangles
216  // are 0.
217  if ((visibility_bits[0] | visibility_bits[1] | visibility_bits[2]) == 0)
218  {
219  cpp17::optional<detail::TriangleToRasterize> t = detail::process_prospective_tri(
220  clipspace_vertices[tri_indices[0]], clipspace_vertices[tri_indices[1]],
221  clipspace_vertices[tri_indices[2]], viewport_width, viewport_height, enable_backface_culling);
222  if (t)
223  {
224  triangles_to_raster.push_back(*t);
225  }
226  continue;
227  }
228  // at this moment the triangle is known to be intersecting one of the view frustum's planes
229  std::vector<detail::Vertex<float>> vertices;
230  vertices.push_back(clipspace_vertices[tri_indices[0]]);
231  vertices.push_back(clipspace_vertices[tri_indices[1]]);
232  vertices.push_back(clipspace_vertices[tri_indices[2]]);
233  // split the triangle if it intersects the near plane:
234  if (enable_near_clipping)
235  {
236  vertices = detail::clip_polygon_to_plane_in_4d(
237  vertices, glm::tvec4<float>(0.0f, 0.0f, -1.0f, -1.0f)); // "Normal" (or "4D hyperplane") of
238  // the near-plane. I tested it and it
239  // works like this but I'm a little
240  // bit unsure because Songho says the
241  // normal of the near-plane is
242  // (0,0,-1,1) (maybe I have to switch
243  // around the < 0 checks in the
244  // function?)
245  }
246 
247  // triangulation of the polygon formed of vertices array
248  if (vertices.size() >= 3)
249  {
250  for (unsigned char k = 0; k < vertices.size() - 2; k++)
251  {
252  cpp17::optional<detail::TriangleToRasterize> t =
253  detail::process_prospective_tri(vertices[0], vertices[1 + k], vertices[2 + k],
254  viewport_width, viewport_height, enable_backface_culling);
255  if (t)
256  {
257  triangles_to_raster.push_back(*t);
258  }
259  }
260  }
261  }
262 
263  // Fragment/pixel shader: Colour the pixel values
264  for (const auto& tri : triangles_to_raster)
265  {
266  detail::raster_triangle(tri, colorbuffer, depthbuffer, texture, enable_far_clipping);
267  }
268  return std::make_pair(colorbuffer, depthbuffer);
269 };
