89 inline std::pair<std::vector<core::Mesh>, std::vector<fitting::RenderingParameters>> fit_shape_and_pose(
90  const morphablemodel::MorphableModel& morphable_model,
91  const std::vector<morphablemodel::Blendshape>& blendshapes,
92  const std::vector<core::LandmarkCollection<Eigen::Vector2f>>& landmarks,
93  const core::LandmarkMapper& landmark_mapper, std::vector<int> image_width, std::vector<int> image_height,
94  const morphablemodel::EdgeTopology& edge_topology, const fitting::ContourLandmarks& contour_landmarks,
95  const fitting::ModelContour& model_contour, int num_iterations,
96  cpp17::optional<int> num_shape_coefficients_to_fit, float lambda,
97  cpp17::optional<fitting::RenderingParameters> initial_rendering_params,
98  std::vector<float>& pca_shape_coefficients, std::vector<std::vector<float>>& blendshape_coefficients,
99  std::vector<std::vector<Eigen::Vector2f>>& fitted_image_points)
100 {
101  assert(blendshapes.size() > 0);
102  assert(landmarks.size() > 0 && landmarks.size() == image_width.size() &&
103  image_width.size() == image_height.size());
104  assert(num_iterations > 0); // Can we allow 0, for only the initial pose-fit?
105  assert(pca_shape_coefficients.size() <= morphable_model.get_shape_model().get_num_principal_components());
106  int num_images = static_cast<int>(landmarks.size());
107  for (int j = 0; j < num_images; ++j)
108  {
109  assert(landmarks[j].size() >= 4);
110  assert(image_width[j] > 0 && image_height[j] > 0);
111  }
112  // More asserts I forgot?
113 
114  using Eigen::MatrixXf;
115  using Eigen::Vector2f;
116  using Eigen::Vector4f;
117  using Eigen::VectorXf;
118  using std::vector;
119 
120  if (!num_shape_coefficients_to_fit)
121  {
122  num_shape_coefficients_to_fit = morphable_model.get_shape_model().get_num_principal_components();
123  }
124 
125  if (pca_shape_coefficients.empty())
126  {
127  pca_shape_coefficients.resize(num_shape_coefficients_to_fit.value());
128  }
129  // Todo: This leaves the following case open: num_coeffs given is empty or defined, but the
130  // pca_shape_coefficients given is != num_coeffs or the model's max-coeffs. What to do then? Handle &
131  // document!
132 
133  if (blendshape_coefficients.empty())
134  {
135  for (int j = 0; j < num_images; ++j)
136  {
137  std::vector<float> current_blendshape_coefficients;
138  current_blendshape_coefficients.resize(blendshapes.size());
139  blendshape_coefficients.push_back(current_blendshape_coefficients);
140  }
141  }
142 
143  MatrixXf blendshapes_as_basis = morphablemodel::to_matrix(blendshapes);
144 
145  // Current mesh - either from the given coefficients, or the mean:
146  VectorXf current_pca_shape = morphable_model.get_shape_model().draw_sample(pca_shape_coefficients);
147  vector<VectorXf> current_combined_shapes;
148  vector<core::Mesh> current_meshes;
149  for (int j = 0; j < num_images; ++j)
150  {
151  VectorXf current_combined_shape =
152  current_pca_shape +
153  blendshapes_as_basis * Eigen::Map<const Eigen::VectorXf>(blendshape_coefficients[j].data(),
154  blendshape_coefficients[j].size());
155  current_combined_shapes.push_back(current_combined_shape);
156 
157  core::Mesh current_mesh = morphablemodel::sample_to_mesh(
158  current_combined_shape, morphable_model.get_color_model().get_mean(),
159  morphable_model.get_shape_model().get_triangle_list(),
160  morphable_model.get_color_model().get_triangle_list(), morphable_model.get_texture_coordinates());
161  current_meshes.push_back(current_mesh);
162  }
163 
164  // The 2D and 3D point correspondences used for the fitting:
165  vector<vector<Vector4f>> model_points; // the points in the 3D shape model of all frames
166  vector<vector<int>> vertex_indices; // their vertex indices of all frames
167  vector<vector<Vector2f>> image_points; // the corresponding 2D landmark points of all frames
168 
169  for (int j = 0; j < num_images; ++j)
170  {
171  vector<Vector4f> current_model_points;
172  vector<int> current_vertex_indices;
173  vector<Vector2f> current_image_points;
174 
175  // Sub-select all the landmarks which we have a mapping for (i.e. that are defined in the 3DMM),
176  // and get the corresponding model points (mean if given no initial coeffs, from the computed shape
177  // otherwise):
178  for (int i = 0; i < landmarks[j].size(); ++i)
179  {
180  auto converted_name = landmark_mapper.convert(landmarks[j][i].name);
181  if (!converted_name)
182  { // no mapping defined for the current landmark
183  continue;
184  }
185  int vertex_idx = std::stoi(converted_name.value());
186  Vector4f vertex(current_meshes[j].vertices[vertex_idx][0],
187  current_meshes[j].vertices[vertex_idx][1],
188  current_meshes[j].vertices[vertex_idx][2], 1.0f);
189  current_model_points.emplace_back(vertex);
190  current_vertex_indices.emplace_back(vertex_idx);
191  current_image_points.emplace_back(landmarks[j][i].coordinates);
192  }
193 
194  model_points.push_back(current_model_points);
195  vertex_indices.push_back(current_vertex_indices);
196  image_points.push_back(current_image_points);
197  }
198 
199  // Need to do an initial pose fit to do the contour fitting inside the loop.
200  // We'll do an expression fit too, since face shapes vary quite a lot, depending on expressions.
201  vector<fitting::RenderingParameters> rendering_params;
202  for (int j = 0; j < num_images; ++j)
203  {
204  fitting::ScaledOrthoProjectionParameters current_pose =
205  fitting::estimate_orthographic_projection_linear(image_points[j], model_points[j], true,
206  image_height[j]);
207  fitting::RenderingParameters current_rendering_params(current_pose, image_width[j], image_height[j]);
208  rendering_params.push_back(current_rendering_params);
209 
210  const Eigen::Matrix<float, 3, 4> affine_from_ortho =
211  fitting::get_3x4_affine_camera_matrix(current_rendering_params, image_width[j], image_height[j]);
212  blendshape_coefficients[j] = fitting::fit_blendshapes_to_landmarks_nnls(
213  blendshapes, current_pca_shape, affine_from_ortho, image_points[j], vertex_indices[j]);
214 
215  // Mesh with same PCA coeffs as before, but new expression fit (this is relevant if no initial
216  // blendshape coeffs have been given):
217  current_combined_shapes[j] =
218  current_pca_shape + morphablemodel::to_matrix(blendshapes) *
219  Eigen::Map<const Eigen::VectorXf>(blendshape_coefficients[j].data(),
220  blendshape_coefficients[j].size());
221  current_meshes[j] = morphablemodel::sample_to_mesh(
222  current_combined_shapes[j], morphable_model.get_color_model().get_mean(),
223  morphable_model.get_shape_model().get_triangle_list(),
224  morphable_model.get_color_model().get_triangle_list(), morphable_model.get_texture_coordinates());
225  }
226 
227  // The static (fixed) landmark correspondences which will stay the same throughout
228  // the fitting (the inner face landmarks):
229  vector<vector<int>> fixed_vertex_indices(vertex_indices);
230  vector<vector<Vector2f>> fixed_image_points(image_points);
231 
232  for (int i = 0; i < num_iterations; ++i)
233  {
234  vector<Eigen::Matrix<float, 3, 4>> affine_from_orthos;
235  vector<VectorXf> mean_plus_blendshapes;
236 
237  image_points = fixed_image_points;
238  vertex_indices = fixed_vertex_indices;
239 
240  for (int j = 0; j < num_images; ++j)
241  {
242  // Given the current pose, find 2D-3D contour correspondences of the front-facing face contour:
243  vector<Vector2f> image_points_contour;
244  vector<int> vertex_indices_contour;
245  auto yaw_angle = glm::degrees(glm::eulerAngles(rendering_params[j].get_rotation())[1]);
246  // For each 2D contour landmark, get the corresponding 3D vertex point and vertex id:
247  std::tie(image_points_contour, std::ignore, vertex_indices_contour) =
248  fitting::get_contour_correspondences(
249  landmarks[j], contour_landmarks, model_contour, yaw_angle, current_meshes[j],
250  rendering_params[j].get_modelview(), rendering_params[j].get_projection(),
251  fitting::get_opencv_viewport(image_width[j], image_height[j]));
252  // Add the contour correspondences to the set of landmarks that we use for the fitting:
253  vertex_indices[j] = fitting::concat(vertex_indices[j], vertex_indices_contour);
254  image_points[j] = fitting::concat(image_points[j], image_points_contour);
255 
256  // Fit the occluding (away-facing) contour using the detected contour LMs:
257  vector<Vector2f> occluding_contour_landmarks;
258  if (yaw_angle >= 0.0f) // positive yaw = subject looking to the left
259  { // the left contour is the occluding one we want to use ("away-facing")
260  auto contour_landmarks_ = core::filter(
261  landmarks[j], contour_landmarks.left_contour); // Can do this outside of the loop
262  std::for_each(
263  begin(contour_landmarks_), end(contour_landmarks_),
264  [&occluding_contour_landmarks](auto&& lm) {
265  occluding_contour_landmarks.push_back({lm.coordinates[0], lm.coordinates[1]});
266  });
267  } else
268  {
269  auto contour_landmarks_ = core::filter(landmarks[j], contour_landmarks.right_contour);
270  std::for_each(
271  begin(contour_landmarks_), end(contour_landmarks_),
272  [&occluding_contour_landmarks](auto&& lm) {
273  occluding_contour_landmarks.push_back({lm.coordinates[0], lm.coordinates[1]});
274  });
275  }
276  auto edge_correspondences = fitting::find_occluding_edge_correspondences(
277  current_meshes[j], edge_topology, rendering_params[j], occluding_contour_landmarks, 180.0f);
278  image_points[j] = fitting::concat(image_points[j], edge_correspondences.first);
279  vertex_indices[j] = fitting::concat(vertex_indices[j], edge_correspondences.second);
280 
281  // Get the model points of the current mesh, for all correspondences that we've got:
282  model_points[j].clear();
283  for (auto v : vertex_indices[j])
284  {
285  model_points[j].push_back({current_meshes[j].vertices[v][0], current_meshes[j].vertices[v][1],
286  current_meshes[j].vertices[v][2], 1.0f});
287  }
288 
289  // Re-estimate the pose, using all correspondences:
290  fitting::ScaledOrthoProjectionParameters current_pose =
291  fitting::estimate_orthographic_projection_linear(image_points[j], model_points[j], true,
292  image_height[j]);
293  rendering_params[j] = fitting::RenderingParameters(current_pose, image_width[j], image_height[j]);
294 
295  const Eigen::Matrix<float, 3, 4> affine_from_ortho =
296  fitting::get_3x4_affine_camera_matrix(rendering_params[j], image_width[j], image_height[j]);
297  affine_from_orthos.push_back(affine_from_ortho);
298 
299  // Estimate the PCA shape coefficients with the current blendshape coefficients:
300  VectorXf current_mean_plus_blendshapes =
301  morphable_model.get_shape_model().get_mean() +
302  blendshapes_as_basis * Eigen::Map<const Eigen::VectorXf>(blendshape_coefficients[j].data(),
303  blendshape_coefficients[j].size());
304  mean_plus_blendshapes.push_back(current_mean_plus_blendshapes);
305  }
306  pca_shape_coefficients = fitting::fit_shape_to_landmarks_linear_multi(
307  morphable_model.get_shape_model(), affine_from_orthos, image_points, vertex_indices,
308  mean_plus_blendshapes, lambda, num_shape_coefficients_to_fit);
309 
310  // Estimate the blendshape coefficients with the current PCA model estimate:
311  current_pca_shape = morphable_model.get_shape_model().draw_sample(pca_shape_coefficients);
312 
313  for (int j = 0; j < num_images; ++j)
314  {
315  blendshape_coefficients[j] = fitting::fit_blendshapes_to_landmarks_nnls(
316  blendshapes, current_pca_shape, affine_from_orthos[j], image_points[j], vertex_indices[j]);
317 
318  current_combined_shapes[j] =
319  current_pca_shape +
320  blendshapes_as_basis * Eigen::Map<const Eigen::VectorXf>(blendshape_coefficients[j].data(),
321  blendshape_coefficients[j].size());
322  current_meshes[j] = morphablemodel::sample_to_mesh(
323  current_combined_shapes[j], morphable_model.get_color_model().get_mean(),
324  morphable_model.get_shape_model().get_triangle_list(),
325  morphable_model.get_color_model().get_triangle_list(),
326  morphable_model.get_texture_coordinates());
327  }
328  }
329 
330  fitted_image_points = image_points;
331  return {current_meshes, rendering_params}; // I think we could also work with a Mat face_instance in this
332  // function instead of a Mesh, but it would convolute the code
333  // more (i.e. more complicated to access vertices).
334 };
