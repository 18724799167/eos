62 inline std::vector<int> occluding_boundary_vertices(const core::Mesh& mesh,
63  const morphablemodel::EdgeTopology& edge_topology,
64  glm::mat4x4 R, bool perform_self_occlusion_check = true)
65 {
66  // Rotate the mesh:
67  std::vector<glm::vec4> rotated_vertices;
68  std::for_each(begin(mesh.vertices), end(mesh.vertices), [&rotated_vertices, &R](const auto& v) {
69  rotated_vertices.push_back(R * glm::vec4(v[0], v[1], v[2], 1.0f));
70  });
71 
72  // Compute the face normals of the rotated mesh:
73  std::vector<glm::vec3> facenormals;
74  for (const auto& f : mesh.tvi)
75  { // for each face (triangle):
76  const auto n =
77  render::compute_face_normal(glm::vec3(rotated_vertices[f[0]]), glm::vec3(rotated_vertices[f[1]]),
78  glm::vec3(rotated_vertices[f[2]]));
79  facenormals.push_back(n);
80  }
81 
82  // Find occluding edges:
83  std::vector<int> occluding_edges_indices;
84  for (int edge_idx = 0; edge_idx < edge_topology.adjacent_faces.size();
85  ++edge_idx) // For each edge... Ef contains the indices of the two adjacent faces
86  {
87  const auto& edge = edge_topology.adjacent_faces[edge_idx];
88  if (edge[0] == 0) // ==> NOTE/Todo Need to change this if we use 0-based indexing!
89  {
90  // Edges with a zero index lie on the mesh boundary, i.e. they are only
91  // adjacent to one face.
92  continue;
93  }
94  // Compute the occluding edges as those where the two adjacent face normals
95  // differ in the sign of their z-component:
96  // Changing from 1-based indexing to 0-based!
97  if (glm::sign(facenormals[edge[0] - 1].z) != glm::sign(facenormals[edge[1] - 1].z))
98  {
99  // It's an occluding edge, store the index:
100  occluding_edges_indices.push_back(edge_idx);
101  }
102  }
103  // Select the vertices lying at the two ends of the occluding edges and remove duplicates:
104  // (This is what EdgeTopology::adjacent_vertices is needed for).
105  std::vector<int> occluding_vertices; // The model's contour vertices
106  for (auto edge_idx : occluding_edges_indices)
107  {
108  // Changing from 1-based indexing to 0-based!
109  occluding_vertices.push_back(edge_topology.adjacent_vertices[edge_idx][0] - 1);
110  occluding_vertices.push_back(edge_topology.adjacent_vertices[edge_idx][1] - 1);
111  }
112  // Remove duplicate vertex id's (std::unique works only on sorted sequences):
113  std::sort(begin(occluding_vertices), end(occluding_vertices));
114  occluding_vertices.erase(std::unique(begin(occluding_vertices), end(occluding_vertices)),
115  end(occluding_vertices));
116 
117  if (perform_self_occlusion_check)
118  {
119  // Perform ray-casting to find out which vertices are not visible (i.e. self-occluded):
120  std::vector<bool> visibility;
121  const glm::vec3 ray_direction(0.0f, 0.0f,
122  1.0f); // we shoot rays from the vertex towards the camera
123  for (auto vertex_idx : occluding_vertices)
124  {
125  const glm::vec3 ray_origin(rotated_vertices[vertex_idx]);
126  bool visible = true;
127  // For every tri of the rotated mesh:
128  for (const auto& tri : mesh.tvi)
129  {
130  const auto& v0 = rotated_vertices[tri[0]];
131  const auto& v1 = rotated_vertices[tri[1]];
132  const auto& v2 = rotated_vertices[tri[2]];
133 
134  const auto intersect = render::ray_triangle_intersect(
135  ray_origin, ray_direction, glm::vec3(v0), glm::vec3(v1), glm::vec3(v2), false);
136  // first is bool intersect, second is the distance t
137  if (intersect.first == true)
138  {
139  // We've hit a triangle. Ray hit its own triangle. If it's behind the ray origin, ignore
140  // the intersection:
141  // Note: Check if in front or behind?
142  if (intersect.second.value() <= 1e-4)
143  {
144  continue; // the intersection is behind the vertex, we don't care about it
145  }
146  // Otherwise, we've hit a genuine triangle, and the vertex is not visible:
147  visible = false;
148  break;
149  }
150  }
151  visibility.push_back(visible);
152  }
153 
154  // Remove vertices from occluding boundary list that are not visible:
155  std::vector<int> final_vertex_ids;
156  for (int i = 0; i < occluding_vertices.size(); ++i)
157  {
158  if (visibility[i] == true)
159  {
160  final_vertex_ids.push_back(occluding_vertices[i]);
161  }
162  }
163  return final_vertex_ids;
164  } else
165  {
166  return occluding_vertices;
167  }
168 };

187 template <class VectorOfVectorsType, typename num_t = double, int DIM = -1,
188  class Distance = nanoflann::metric_L2, typename IndexType = size_t>
189 struct KDTreeVectorOfVectorsAdaptor
190 {
191  typedef KDTreeVectorOfVectorsAdaptor<VectorOfVectorsType, num_t, DIM, Distance> self_t;
192  typedef typename Distance::template traits<num_t, self_t>::distance_t metric_t;
193  typedef nanoflann::KDTreeSingleIndexAdaptor<metric_t, self_t, DIM, IndexType> index_t;
194 
195  index_t* index;
196 
198  // Make sure the data is kept alive while the kd-tree is in use.
199  KDTreeVectorOfVectorsAdaptor(const int dimensionality, const VectorOfVectorsType& mat,
200  const int leaf_max_size = 10)
201  : m_data(mat)
202  {
203  assert(mat.size() != 0 && mat[0].size() != 0);
204  const size_t dims = mat[0].size();
205  if (DIM > 0 && static_cast<int>(dims) != DIM)
206  throw std::runtime_error("Data set dimensionality does not match the 'DIM' template argument");
207  index = new index_t(dims, *this /* adaptor */, nanoflann::KDTreeSingleIndexAdaptorParams(leaf_max_size));
208  index->buildIndex();
209  }
210 
211  ~KDTreeVectorOfVectorsAdaptor() { delete index; }
212 
213  const VectorOfVectorsType& m_data;

220  inline void query(const num_t* query_point, const size_t num_closest, IndexType* out_indices,
221  num_t* out_distances_sq, const int nChecks_IGNORED = 10) const
222  {
223  nanoflann::KNNResultSet<num_t, IndexType> resultSet(num_closest);
224  resultSet.init(out_indices, out_distances_sq);
225  index->findNeighbors(resultSet, query_point, nanoflann::SearchParams());
226  }

231  const self_t& derived() const { return *this; }
232  self_t& derived() { return *this; }
233 
234  // Must return the number of data points
235  inline size_t kdtree_get_point_count() const { return m_data.size(); }
236 
237  // Returns the distance between the vector "p1[0:size-1]" and the data point with index "idx_p2" stored in
238  // the class:
239  inline num_t kdtree_distance(const num_t* p1, const size_t idx_p2, size_t size) const
240  {
241  num_t s = 0;
242  for (size_t i = 0; i < size; i++)
243  {
244  const num_t d = p1[i] - m_data[idx_p2][i];
245  s += d * d;
246  }
247  return s;
248  }
249 
250  // Returns the dim'th component of the idx'th point in the class:
251  inline num_t kdtree_get_pt(const size_t idx, int dim) const
252  {
253  return m_data[idx][dim];
254  }
255 
256  // Optional bounding-box computation: return false to default to a standard bbox computation loop.
257  // Return true if the BBOX was already computed by the class and returned in "bb" so it can be avoided to redo it again.
258  // Look at bb.size() to find out the expected dimensionality (e.g. 2 or 3 for point clouds)
259  template <class BBOX>
260  bool kdtree_get_bbox(BBOX& /*bb*/) const
261  {
262  return false;
263  }

295 inline std::pair<std::vector<Eigen::Vector2f>, std::vector<int>> find_occluding_edge_correspondences(
296  const core::Mesh& mesh, const morphablemodel::EdgeTopology& edge_topology,
297  const fitting::RenderingParameters& rendering_parameters, const std::vector<Eigen::Vector2f>& image_edges,
298  float distance_threshold = 64.0f, bool perform_self_occlusion_check = true)
299 {
300  assert(rendering_parameters.get_camera_type() == fitting::CameraType::Orthographic);
301  using Eigen::Vector2f;
302  using std::vector;
303 
304  // If there are no image_edges given, there's no point in computing anything:
305  if (image_edges.empty())
306  {
307  return {};
308  }
309 
310  // Compute vertices that lye on occluding boundaries:
311  const auto occluding_vertices = occluding_boundary_vertices(
312  mesh, edge_topology, glm::mat4x4(rendering_parameters.get_rotation()), perform_self_occlusion_check);
313 
314  // Project these occluding boundary vertices from 3D to 2D:
315  vector<Vector2f> model_edges_projected;
316  for (const auto& v : occluding_vertices)
317  {
318  const auto p =
319  glm::project({mesh.vertices[v][0], mesh.vertices[v][1], mesh.vertices[v][2]},
320  rendering_parameters.get_modelview(), rendering_parameters.get_projection(),
321  fitting::get_opencv_viewport(rendering_parameters.get_screen_width(),
322  rendering_parameters.get_screen_height()));
323  model_edges_projected.push_back({p.x, p.y});
324  }
325 
326  // Find edge correspondences:
327  // Build a kd-tree and use nearest neighbour search:
328  using kd_tree_t = KDTreeVectorOfVectorsAdaptor<vector<Vector2f>, float, 2>;
329  kd_tree_t tree(2, image_edges); // dim, samples, max_leaf
330  tree.index->buildIndex();
331 
332  vector<std::pair<std::size_t, double>>
333  idx_d; // will contain [ index to the 2D edge in 'image_edges', distance (L2^2) ]
334  for (const auto& e : model_edges_projected)
335  {
336  std::size_t idx; // contains the indices into the original 'image_edges' vector
337  double dist_sq; // squared distances
338  nanoflann::KNNResultSet<double> resultSet(1);
339  resultSet.init(&idx, &dist_sq);
340  tree.index->findNeighbors(resultSet, e.data(), nanoflann::SearchParams(10));
341  idx_d.push_back({idx, dist_sq});
342  }
343  // Filter edge matches:
344  // We filter below by discarding all correspondence that are a certain distance apart.
345  // We could also (or in addition to) discard the worst 5% of the distances or something like that.
346 
347  // Filter and store the image (edge) points with their corresponding vertex id:
348  vector<int> vertex_indices;
349  vector<Vector2f> image_points;
350  assert(occluding_vertices.size() == idx_d.size());
351  for (int i = 0; i < occluding_vertices.size(); ++i)
352  {
353  const auto ortho_scale =
354  rendering_parameters.get_screen_width() /
355  rendering_parameters.get_frustum().r; // This might be a bit of a hack - we recover the "real"
356  // scaling from the SOP estimate
357  if (idx_d[i].second <= distance_threshold * ortho_scale) // I think multiplying by the scale is good
358  // here and gives us invariance w.r.t. the
359  // image resolution and face size.
360  {
361  const auto edge_point = image_edges[idx_d[i].first];
362  // Store the found 2D edge point, and the associated vertex id:
363  vertex_indices.push_back(occluding_vertices[i]);
364  image_points.push_back(edge_point);
365  }
366  }
367  return {image_points, vertex_indices};
368 };
