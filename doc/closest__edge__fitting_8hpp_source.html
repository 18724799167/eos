69 inline std::pair<bool, cpp17::optional<float>>
70 ray_triangle_intersect(const glm::vec3& ray_origin, const glm::vec3& ray_direction, const glm::vec3& v0,
71  const glm::vec3& v1, const glm::vec3& v2, bool enable_backculling)
72 {
73  using glm::vec3;
74  const float epsilon = 1e-6f;
75 
76  const vec3 v0v1 = v1 - v0;
77  const vec3 v0v2 = v2 - v0;
78 
79  const vec3 pvec = glm::cross(ray_direction, v0v2);
80 
81  const float det = glm::dot(v0v1, pvec);
82  if (enable_backculling)
83  {
84  // If det is negative, the triangle is back-facing.
85  // If det is close to 0, the ray misses the triangle.
86  if (det < epsilon)
87  return {false, cpp17::nullopt};
88  } else
89  {
90  // If det is close to 0, the ray and triangle are parallel.
91  if (std::abs(det) < epsilon)
92  return {false, cpp17::nullopt};
93  }
94  const float inv_det = 1 / det;
95 
96  const vec3 tvec = ray_origin - v0;
97  const auto u = glm::dot(tvec, pvec) * inv_det;
98  if (u < 0 || u > 1)
99  return {false, cpp17::nullopt};
100 
101  const vec3 qvec = glm::cross(tvec, v0v1);
102  const auto v = glm::dot(ray_direction, qvec) * inv_det;
103  if (v < 0 || u + v > 1)
104  return {false, cpp17::nullopt};
105 
106  const auto t = glm::dot(v0v2, qvec) * inv_det;
107 
108  return {true, t};
109 };

124 inline std::vector<int> occluding_boundary_vertices(const core::Mesh& mesh,
125  const morphablemodel::EdgeTopology& edge_topology,
126  glm::mat4x4 R)
127 {
128  // Rotate the mesh:
129  std::vector<glm::vec4> rotated_vertices;
130  std::for_each(begin(mesh.vertices), end(mesh.vertices), [&rotated_vertices, &R](const auto& v) {
131  rotated_vertices.push_back(R * glm::vec4(v[0], v[1], v[2], 1.0f));
132  });
133 
134  // Compute the face normals of the rotated mesh:
135  std::vector<glm::vec3> facenormals;
136  for (const auto& f : mesh.tvi)
137  { // for each face (triangle):
138  const auto n =
139  render::compute_face_normal(glm::vec3(rotated_vertices[f[0]]), glm::vec3(rotated_vertices[f[1]]),
140  glm::vec3(rotated_vertices[f[2]]));
141  facenormals.push_back(n);
142  }
143 
144  // Find occluding edges:
145  std::vector<int> occluding_edges_indices;
146  for (int edge_idx = 0; edge_idx < edge_topology.adjacent_faces.size();
147  ++edge_idx) // For each edge... Ef contains the indices of the two adjacent faces
148  {
149  const auto& edge = edge_topology.adjacent_faces[edge_idx];
150  if (edge[0] == 0) // ==> NOTE/Todo Need to change this if we use 0-based indexing!
151  {
152  // Edges with a zero index lie on the mesh boundary, i.e. they are only
153  // adjacent to one face.
154  continue;
155  }
156  // Compute the occluding edges as those where the two adjacent face normals
157  // differ in the sign of their z-component:
158  // Changing from 1-based indexing to 0-based!
159  if (glm::sign(facenormals[edge[0] - 1].z) != glm::sign(facenormals[edge[1] - 1].z))
160  {
161  // It's an occluding edge, store the index:
162  occluding_edges_indices.push_back(edge_idx);
163  }
164  }
165  // Select the vertices lying at the two ends of the occluding edges and remove duplicates:
166  // (This is what EdgeTopology::adjacent_vertices is needed for).
167  std::vector<int> occluding_vertices; // The model's contour vertices
168  for (auto edge_idx : occluding_edges_indices)
169  {
170  // Changing from 1-based indexing to 0-based!
171  occluding_vertices.push_back(edge_topology.adjacent_vertices[edge_idx][0] - 1);
172  occluding_vertices.push_back(edge_topology.adjacent_vertices[edge_idx][1] - 1);
173  }
174  // Remove duplicate vertex id's (std::unique works only on sorted sequences):
175  std::sort(begin(occluding_vertices), end(occluding_vertices));
176  occluding_vertices.erase(std::unique(begin(occluding_vertices), end(occluding_vertices)),
177  end(occluding_vertices));
178 
179  // Perform ray-casting to find out which vertices are not visible (i.e. self-occluded):
180  std::vector<bool> visibility;
181  for (auto vertex_idx : occluding_vertices)
182  {
183  bool visible = true;
184  // For every tri of the rotated mesh:
185  for (const auto& tri : mesh.tvi)
186  {
187  const auto& v0 = rotated_vertices[tri[0]];
188  const auto& v1 = rotated_vertices[tri[1]];
189  const auto& v2 = rotated_vertices[tri[2]];
190 
191  const glm::vec3 ray_origin(rotated_vertices[vertex_idx]);
192  const glm::vec3 ray_direction(0.0f, 0.0f, 1.0f); // we shoot the ray from the vertex towards the camera
193  const auto intersect = ray_triangle_intersect(ray_origin, ray_direction, glm::vec3(v0),
194  glm::vec3(v1), glm::vec3(v2), false);
195  // first is bool intersect, second is the distance t
196  if (intersect.first == true)
197  {
198  // We've hit a triangle. Ray hit its own triangle. If it's behind the ray origin, ignore the intersection:
199  // Check if in front or behind?
200  if (intersect.second.value() <= 1e-4)
201  {
202  continue; // the intersection is behind the vertex, we don't care about it
203  }
204  // Otherwise, we've hit a genuine triangle, and the vertex is not visible:
205  visible = false;
206  break;
207  }
208  }
209  visibility.push_back(visible);
210  }
211 
212  // Remove vertices from occluding boundary list that are not visible:
213  std::vector<int> final_vertex_ids;
214  for (int i = 0; i < occluding_vertices.size(); ++i)
215  {
216  if (visibility[i] == true)
217  {
218  final_vertex_ids.push_back(occluding_vertices[i]);
219  }
220  }
221  return final_vertex_ids;
222 };

243 struct KDTreeVectorOfVectorsAdaptor

252  // Make sure the data is kept alive while the kd-tree is in use.
253  KDTreeVectorOfVectorsAdaptor(const int dimensionality, const VectorOfVectorsType& mat,

275  num_t* out_distances_sq, const int nChecks_IGNORED = 10) const
276  {
277  nanoflann::KNNResultSet<num_t, IndexType> resultSet(num_closest);

279  index->findNeighbors(resultSet, query_point, nanoflann::SearchParams());
280  }

286  self_t& derived() { return *this; }

288  // Must return the number of data points
289  inline size_t kdtree_get_point_count() const { return m_data.size(); }

291  // Returns the distance between the vector "p1[0:size-1]" and the data point with index "idx_p2" stored in
292  // the class:
293  inline num_t kdtree_distance(const num_t* p1, const size_t idx_p2, size_t size) const
294  {
295  num_t s = 0;
296  for (size_t i = 0; i < size; i++)
297  {
298  const num_t d = p1[i] - m_data[idx_p2][i];
299  s += d * d;
300  }
301  return s;
302  }
303 
304  // Returns the dim'th component of the idx'th point in the class:
305  inline num_t kdtree_get_pt(const size_t idx, int dim) const
306  {
307  return m_data[idx][dim];
308  }
309 
310  // Optional bounding-box computation: return false to default to a standard bbox computation loop.
311  // Return true if the BBOX was already computed by the class and returned in "bb" so it can be avoided to redo it again.
312  // Look at bb.size() to find out the expected dimensionality (e.g. 2 or 3 for point clouds)
313  template <class BBOX>
314  bool kdtree_get_bbox(BBOX& /*bb*/) const
315  {
316  return false;
317  }

348 inline std::pair<std::vector<Eigen::Vector2f>, std::vector<int>>
349 find_occluding_edge_correspondences(const core::Mesh& mesh, const morphablemodel::EdgeTopology& edge_topology,
350  const fitting::RenderingParameters& rendering_parameters,
351  const std::vector<Eigen::Vector2f>& image_edges,
352  float distance_threshold = 64.0f)
353 {
354  assert(rendering_parameters.get_camera_type() == fitting::CameraType::Orthographic);
355  using Eigen::Vector2f;
356  using std::vector;
357 
358  // If there are no image_edges given, there's no point in computing anything:
359  if (image_edges.empty())
360  {
361  return {};
362  }
363 
364  // Compute vertices that lye on occluding boundaries:
365  const auto occluding_vertices =
366  occluding_boundary_vertices(mesh, edge_topology, glm::mat4x4(rendering_parameters.get_rotation()));
367 
368  // Project these occluding boundary vertices from 3D to 2D:
369  vector<Vector2f> model_edges_projected;
370  for (const auto& v : occluding_vertices)
371  {
372  const auto p =
373  glm::project({mesh.vertices[v][0], mesh.vertices[v][1], mesh.vertices[v][2]},
374  rendering_parameters.get_modelview(), rendering_parameters.get_projection(),
375  fitting::get_opencv_viewport(rendering_parameters.get_screen_width(),
376  rendering_parameters.get_screen_height()));
377  model_edges_projected.push_back({p.x, p.y});
378  }
379 
380  // Find edge correspondences:
381  // Build a kd-tree and use nearest neighbour search:
382  using kd_tree_t = KDTreeVectorOfVectorsAdaptor<vector<Vector2f>, float, 2>;
383  kd_tree_t tree(2, image_edges); // dim, samples, max_leaf
384  tree.index->buildIndex();
385 
386  vector<std::pair<std::size_t, double>>
387  idx_d; // will contain [ index to the 2D edge in 'image_edges', distance (L2^2) ]
388  for (const auto& e : model_edges_projected)
389  {
390  std::size_t idx; // contains the indices into the original 'image_edges' vector
391  double dist_sq; // squared distances
392  nanoflann::KNNResultSet<double> resultSet(1);
393  resultSet.init(&idx, &dist_sq);
394  tree.index->findNeighbors(resultSet, e.data(), nanoflann::SearchParams(10));
395  idx_d.push_back({idx, dist_sq});
396  }
397  // Filter edge matches:
398  // We filter below by discarding all correspondence that are a certain distance apart.
399  // We could also (or in addition to) discard the worst 5% of the distances or something like that.
400 
401  // Filter and store the image (edge) points with their corresponding vertex id:
402  vector<int> vertex_indices;
403  vector<Vector2f> image_points;
404  assert(occluding_vertices.size() == idx_d.size());
405  for (int i = 0; i < occluding_vertices.size(); ++i)
406  {
407  const auto ortho_scale =
408  rendering_parameters.get_screen_width() /
409  rendering_parameters.get_frustum().r; // This might be a bit of a hack - we recover the "real"
410  // scaling from the SOP estimate
411  if (idx_d[i].second <= distance_threshold * ortho_scale) // I think multiplying by the scale is good
412  // here and gives us invariance w.r.t. the
413  // image resolution and face size.
414  {
415  const auto edge_point = image_edges[idx_d[i].first];
416  // Store the found 2D edge point, and the associated vertex id:
417  vertex_indices.push_back(occluding_vertices[i]);
418  image_points.push_back(edge_point);
419  }
420  }
421  return {image_points, vertex_indices};
422 };
