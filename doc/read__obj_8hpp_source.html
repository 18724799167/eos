66 template <class ContainerType>
67 void tokenize(const std::string& str, ContainerType& tokens, const std::string& delimiters = " ",
68  bool trim_empty = false)
69 {
70  std::string::size_type pos, last_pos = 0;
71  const auto length = str.length();
72 
73  using value_type = typename ContainerType::value_type;
74  using size_type = typename ContainerType::size_type;
75 
76  while (last_pos < length + 1)
77  {
78  pos = str.find_first_of(delimiters, last_pos);
79  if (pos == std::string::npos)
80  {
81  pos = length;
82  }
83 
84  if (pos != last_pos || !trim_empty)
85  tokens.push_back(value_type(str.data() + last_pos, (size_type)pos - last_pos));
86 
87  last_pos = pos + 1;
88  }
89 }

134 // Note: Indices in obj's start at 1!
135 // Oh no, these contain 3 entries for triangles, but can be quads (=4 entries), and potentially more.
136 // Their triplets of: vertex, texture and normal indices. Some of them can be missing. E.g.:
137 // f 1 2 3
138 // f 3/1 4/2 5/3
139 // f 6/4/1 3/5/3 7/6/5
140 // f 7//1 8//2 9//3
141 inline auto parse_face(const std::string& line)
142 {
143  using std::string;
144  using std::vector;
145 
146  // Obj indices are 1-based.
147  vector<int> vertex_indices; // size() = 3 or 4
148  vector<int> texture_indices; // size() = 3 or 4
149  vector<int> normal_indices; // size() = 3 or 4
150 
151  vector<string> tokens;
152  tokenize(line, tokens, " ");
153  assert(tokens.size() == 3 || tokens.size() == 4); // For now we need this to be 3 (triangles) or 4 (quads).
154  // Now for each of these tokens, we want to split on "/":
155  for (const auto& token : tokens)
156  {
157  vector<string> subtokens;
158  tokenize(token, subtokens, "/"); // do we want trim_empty true or false?
159  assert(subtokens.size() > 0 && subtokens.size() <= 3); // <= 3 correct or not?
160  // Ok, let's make our life easy, for now only support the 1/2/3 syntax of the FaceWarehouse scans. In
161  // fact the normal_indices are 0... we should check for that - zero-index = ignore, but note that, it's
162  // probably a non-standard obj format extension.
163  assert(subtokens.size() == 3); // FaceWarehouse
164  vertex_indices.push_back(std::stoi(subtokens[0]) - 1); // obj indices are 1-based, so we subtract one.
165  texture_indices.push_back(std::stoi(subtokens[1]) - 1);
166  // subtokens[2] is zero, hence, no normal_indices.
167  }
168 
169  return std::make_tuple(vertex_indices, texture_indices, normal_indices);
170 };

183 inline Mesh read_obj(std::string filename)
184 {
185  std::ifstream file(filename);
186  if (!file)
187  {
188  throw std::runtime_error(std::string("Could not open obj file: " + filename));
189  }
190 
191  // We'll need these helper functions for the parsing:
192  const auto starts_with = [](const std::string& input, const std::string& match) {
193  return input.size() >= match.size() && std::equal(match.begin(), match.end(), input.begin());
194  };
195 
196  /* auto trim_left = [](const std::string& input, std::string pattern = " \t") {
197  auto first = input.find_first_not_of(pattern);
198  if (first == std::string::npos)
199  {
200  return input;
201  }
202  return input.substr(first, input.size());
203  }; */
204 
205  Mesh mesh;
206 
207  std::string line;
208  while (getline(file, line))
209  {
210  if (starts_with(line, "#"))
211  {
212  continue;
213  }
214 
215  if (starts_with(line, "v "))
216  { // matching with a space so that it doesn't match 'vt'
217  auto vertex_data =
218  detail::parse_vertex(line.substr(2)); // pass the string without the first two characters
219  mesh.vertices.push_back(
220  Eigen::Vector3f(vertex_data.first[0], vertex_data.first[1], vertex_data.first[2]));
221  if (vertex_data.second)
222  { // there are vertex colours:
223  mesh.colors.push_back(vertex_data.second.value());
224  }
225  }
226  if (starts_with(line, "vt "))
227  {
228  const auto texcoords = detail::parse_texcoords(line.substr(3));
229  mesh.texcoords.push_back(texcoords);
230  }
231  if (starts_with(line, "vn "))
232  {
233  // detail::parse_vertex_normal(line.substr(3));
234  // Not handled yet, our Mesh class doesn't contain normals right now anyway.
235  }
236  // There's other things like "vp ", which we don't handle
237  if (starts_with(line, "f "))
238  {
239  auto face_data = detail::parse_face(line.substr(2));
240  if (std::get<0>(face_data).size() == 3) // 3 triangle indices, nothing to do:
241  {
242  mesh.tvi.push_back(
243  {std::get<0>(face_data)[0], std::get<0>(face_data)[1], std::get<0>(face_data)[2]});
244  }
245  // If their sizes are 4, we convert the quad to two triangles:
246  // Actually I think MeshLab does the same, it shows the FaceWarehouse number of "Faces" as twice the "f" entries in the obj.
247  else if (std::get<0>(face_data).size() == 4)
248  {
249  // Just create two faces with (quad[0], quad[1], quad[2]) and (quad[0], quad[2], quad[3]).
250  mesh.tvi.push_back(
251  {std::get<0>(face_data)[0], std::get<0>(face_data)[1], std::get<0>(face_data)[2]});
252  mesh.tvi.push_back(
253  {std::get<0>(face_data)[0], std::get<0>(face_data)[2], std::get<0>(face_data)[3]});
254  }
255  // We don't handle normal_indices for now.
256  }
257  // There can be other stuff in obj's like materials, named objects, etc., which are not handled here.
258  }
259  return mesh;
260 }
