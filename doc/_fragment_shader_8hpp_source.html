32 // Fragment shaders are a more accurate name for the same functionality as Pixel shaders. They aren't pixels
33 // yet, since the output still has to past several tests (depth, alpha, stencil) as well as the fact that one
34 // may be using anti-aliasing, which renders one-fragment-to-one-pixel non-true.
35 // The "pixel" in "pixel shader" is a misnomer because the pixel shader doesn't operate on pixels directly.
36 // The pixel shader operates on "fragments" which may or may not end up as actual pixels, depending on several
37 // factors outside of the pixel shader.
38 // The shaders *can not* depend on any state - they have to be able to run independently and in parallel!
39 // But can they really? What about the z-test? It happens earlier at the moment - which is good?

65  template <typename T, glm::precision P = glm::defaultp>
66  glm::tvec4<T, P> shade_triangle_pixel(int x, int y, const detail::Vertex<T, P>& point_a,
67  const detail::Vertex<T, P>& point_b,

69  const cpp17::optional<Texture>& texture, float dudx, float dudy,

110  // The Texture is in BGR, thus tex2D returns BGR

112  // tex2d divides the colour values by 255, so that the return value we get here is in the range [0, 1].
113  glm::tvec3<T, P> texture_color =
114  detail::tex2d(texcoords_persp, texture.value(), dudx, dudy, dvdx, dvdy); // uses the current texture
115  glm::tvec3<T, P> pixel_color = glm::tvec3<T, P>(texture_color[2], texture_color[1], texture_color[0]);
116  // other: color.mul(tex2D(texture, texCoord));
117  return glm::tvec4<T, P>(pixel_color, T(1));

132 template <typename T, glm::precision P = glm::defaultp>
133 glm::tvec3<T, P> compute_inverse_perspectively_correct_lambda(const glm::tvec3<T, P>& lambda_world,
134  const T& one_over_w0, const T& one_over_w1,
135  const T& one_over_w2)
136 {
137  float w0 = 1 / one_over_w0;
138  float w1 = 1 / one_over_w1;
139  float w2 = 1 / one_over_w2;
140 
141  float d = w0 - (w0 - w1) * lambda_world.y - (w0 - w2) * lambda_world.z;
142  if (d == 0)
143  return lambda_world;
144 
145  glm::tvec3<T, P> lambda;
146 
147  lambda.z = lambda_world.z * w2 / d;
148  lambda.y = lambda_world.y * w1 / d;
149 
150  lambda.x = 1 - lambda.y - lambda.z;
151  return lambda;
152 };

156 public:
170  template <typename T, glm::precision P = glm::defaultp>
171  glm::tvec4<T, P> shade_triangle_pixel(int x, int y, const detail::Vertex<T, P>& point_a,
172  const detail::Vertex<T, P>& point_b,
173  const detail::Vertex<T, P>& point_c, const glm::tvec3<T, P>& lambda,
174  const cpp17::optional<Texture>& texture, float dudx, float dudy,
175  float dvdx, float dvdy)
176  {
177  auto corrected_lambda = compute_inverse_perspectively_correct_lambda(
178  lambda, point_a.position.w, point_b.position.w, point_c.position.w);
179  glm::tvec2<T, P> texcoords_persp = corrected_lambda[0] * point_a.texcoords +
180  corrected_lambda[1] * point_b.texcoords +
181  corrected_lambda[2] * point_c.texcoords;
182 
183  // Texturing, no mipmapping:
184  glm::vec2 image_tex_coords = detail::texcoord_wrap(texcoords_persp);
185  image_tex_coords[0] *= texture->mipmaps[0].width();
186  image_tex_coords[1] *= texture->mipmaps[0].height();
187  glm::vec3 texture_color = detail::tex2d_linear(image_tex_coords, 0, texture.value()) / 255.0f;
188  glm::tvec3<T, P> pixel_color = glm::tvec3<T, P>(texture_color[2], texture_color[1], texture_color[0]);
189  return glm::tvec4<T, P>(pixel_color, T(1));
190  };
191 };
