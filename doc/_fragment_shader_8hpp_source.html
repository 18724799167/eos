33 // Fragment shaders are a more accurate name for the same functionality as Pixel shaders. They aren't pixels
34 // yet, since the output still has to past several tests (depth, alpha, stencil) as well as the fact that one
35 // may be using antialiasing, which renders one-fragment-to-one-pixel non-true.
36 // The "pixel" in "pixel shader" is a misnomer because the pixel shader doesn't operate on pixels directly.
37 // The pixel shader operates on "fragments" which may or may not end up as actual pixels, depending on several
38 // factors outside of the pixel shader.
39 // The shaders *can not* depend on any state - they have to be able to run independently and in parallel!
40 // But can they really? What about the z-test? It happens earlier at the moment - which is good?

102  {
103  glm::tvec2<T, P> texcoords_persp =
104  lambda[0] * point_a.texcoords + lambda[1] * point_b.texcoords + lambda[2] * point_c.texcoords;

106  // The Texture is in BGR, thus tex2D returns BGR

110  glm::tvec3<T, P> pixel_color = glm::tvec3<T, P>(texture_color[2], texture_color[1], texture_color[0]);
111  // other: color.mul(tex2D(texture, texCoord));
112  return glm::tvec4<T, P>(pixel_color, T(1));
113  };

127 template <typename T, glm::precision P = glm::defaultp>
128 glm::tvec3<T, P> compute_inverse_perspectively_correct_lambda(const glm::tvec3<T, P>& lambda_world,
129  const T& one_over_w0, const T& one_over_w1,
130  const T& one_over_w2)
131 {
132  float w0 = 1 / one_over_w0;
133  float w1 = 1 / one_over_w1;
134  float w2 = 1 / one_over_w2;
135 
136  float d = w0 - (w0 - w1) * lambda_world.y - (w0 - w2) * lambda_world.z;
137  if (d == 0)
138  return lambda_world;
139 
140  glm::tvec3<T, P> lambda;
141 
142  lambda.z = lambda_world.z * w2 / d;
143  lambda.y = lambda_world.y * w1 / d;
144 
145  lambda.x = 1 - lambda.y - lambda.z;
146  return lambda;
147 };

149 class ExtractionFragmentShader
150 {
151 public:
165  template <typename T, glm::precision P = glm::defaultp>
166  glm::tvec4<T, P> shade_triangle_pixel(int x, int y, const detail::Vertex<T, P>& point_a,
167  const detail::Vertex<T, P>& point_b,
168  const detail::Vertex<T, P>& point_c, const glm::tvec3<T, P>& lambda,
169  const boost::optional<Texture>& texture, float dudx, float dudy,
170  float dvdx, float dvdy)
171  {
172  auto corrected_lambda = compute_inverse_perspectively_correct_lambda(
173  lambda, point_a.position.w, point_b.position.w, point_c.position.w);
174  glm::tvec2<T, P> texcoords_persp = corrected_lambda[0] * point_a.texcoords +
175  corrected_lambda[1] * point_b.texcoords +
176  corrected_lambda[2] * point_c.texcoords;
177 
178  // Texturing, no mipmapping:
179  cv::Vec2f image_tex_coords = detail::texcoord_wrap(cv::Vec2f(texcoords_persp.s, texcoords_persp.t));
180  image_tex_coords[0] *= texture->mipmaps[0].cols;
181  image_tex_coords[1] *= texture->mipmaps[0].rows;
182  cv::Vec3f texture_color = detail::tex2d_linear(image_tex_coords, 0, texture.get()) / 255.0;
183  glm::tvec3<T, P> pixel_color = glm::tvec3<T, P>(texture_color[2], texture_color[1], texture_color[0]);
184  return glm::tvec4<T, P>(pixel_color, T(1));
185  };
186 };
