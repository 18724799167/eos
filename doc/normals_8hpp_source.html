122 inline std::vector<Eigen::Vector3f>
123 compute_vertex_normals(const std::vector<Eigen::Vector3f>& vertices,
124  const std::vector<std::array<int, 3>>& triangle_vertex_indices,
125  const std::vector<Eigen::Vector3f>& face_normals)
126 {
127  std::vector<Eigen::Vector3f> per_vertex_normals;
128  // Initialise with zeros:
129  for (int i = 0; i < vertices.size(); ++i)
130  {
131  per_vertex_normals.emplace_back(Eigen::Vector3f(0.0f, 0.0f, 0.0f));
132  }
133 
134  // Loop over the faces again:
135  for (int i = 0; i < triangle_vertex_indices.size(); ++i)
136  {
137  const auto& tvi = triangle_vertex_indices[i];
138  // Throw normal at each corner:
139  per_vertex_normals[tvi[0]] += face_normals[i];
140  per_vertex_normals[tvi[1]] += face_normals[i];
141  per_vertex_normals[tvi[2]] += face_normals[i];
142  }
143 
144  // Take average via normalization:
145  for (auto& n : per_vertex_normals)
146  {
147  n.normalize();
148  }
149  return per_vertex_normals;
150 };
