61 inline MorphableModel load_scm_model(std::string model_filename, std::string isomap_file = std::string())
62 {
63  using cv::Mat;
64  if (sizeof(unsigned int) != 4) // note/todo: maybe use uint32 or similar instead? Yep, but still we could encounter endianness-trouble.
65  {
66  std::cout << "Warning: We're reading 4 Bytes from the file but sizeof(unsigned int) != 4. Check the code/behaviour." << std::endl;
67  }
68  if (sizeof(double) != 8)
69  {
70  std::cout << "Warning: We're reading 8 Bytes from the file but sizeof(double) != 8. Check the code/behaviour." << std::endl;
71  }
72 
73  std::ifstream model_file(model_filename, std::ios::binary);
74  if (!model_file)
75  {
76  const std::string msg("Unable to open model file: " + model_filename);
77  std::cout << msg << std::endl;
78  throw std::runtime_error(msg);
79  }

81  // Reading the shape model
82  // Read (reference?) num triangles and vertices
83  unsigned int num_vertices = 0;
84  unsigned int num_triangles = 0;
85  model_file.read(reinterpret_cast<char*>(&num_vertices),
86  4); // 1 char = 1 byte. uint32=4bytes. float64=8bytes.
87  model_file.read(reinterpret_cast<char*>(&num_triangles), 4);
88 
89  // Read triangles
90  std::vector<std::array<int, 3>> triangle_list;
91 
92  triangle_list.resize(num_triangles);
93  unsigned int v0, v1, v2;
94  for (unsigned int i = 0; i < num_triangles; ++i)
95  {
96  v0 = v1 = v2 = 0;
97  model_file.read(reinterpret_cast<char*>(&v0), 4); // would be nice to pass a &vector and do it in one
98  model_file.read(reinterpret_cast<char*>(&v1), 4); // go, but didn't work. Maybe a cv::Mat would work?
99  model_file.read(reinterpret_cast<char*>(&v2), 4);
100  triangle_list[i][0] = v0;
101  triangle_list[i][1] = v1;
102  triangle_list[i][2] = v2;
103  }
104 
105  // Read number of rows and columns of the shape projection matrix (pcaBasis)
106  unsigned int num_shape_pca_coeffs = 0;
107  unsigned int num_shape_dims = 0; // dimension of the shape vector (3*num_vertices)
108  model_file.read(reinterpret_cast<char*>(&num_shape_pca_coeffs), 4);
109  model_file.read(reinterpret_cast<char*>(&num_shape_dims), 4);
110 
111  if (3 * num_vertices != num_shape_dims)
112  {
113  std::cout << "Warning: Number of shape dimensions is not equal to three times the number of "
114  "vertices. Something will probably go wrong during the loading."
115  << std::endl;
116  }
117 
118  // Read shape projection matrix
119  Mat orthonormal_pca_basis_shape(num_shape_dims, num_shape_pca_coeffs,
120  CV_32FC1); // m x n (rows x cols) = numShapeDims x numShapePcaCoeffs
121  std::cout << "Loading shape PCA basis matrix with " << orthonormal_pca_basis_shape.rows << " rows and "
122  << orthonormal_pca_basis_shape.cols << " cols." << std::endl;
123  for (unsigned int col = 0; col < num_shape_pca_coeffs; ++col)
124  {
125  for (unsigned int row = 0; row < num_shape_dims; ++row)
126  {
127  double var = 0.0;
128  model_file.read(reinterpret_cast<char*>(&var), 8);
129  orthonormal_pca_basis_shape.at<float>(row, col) = static_cast<float>(var);
130  }
131  }
132 
133  // Read mean shape vector
134  unsigned int mean_dims = 0; // dimension of the mean (3*num_vertices)
135  model_file.read(reinterpret_cast<char*>(&mean_dims), 4);
136  if (mean_dims != num_shape_dims)
137  {
138  std::cout << "Warning: Number of shape dimensions is not equal to the number of dimensions of the "
139  "mean. Something will probably go wrong during the loading."
140  << std::endl;
141  }
142  Mat mean_shape(mean_dims, 1, CV_32FC1);
143  unsigned int counter = 0;
144  double vd0, vd1, vd2;
145  for (unsigned int i = 0; i < mean_dims / 3; ++i)
146  {

148  model_file.read(reinterpret_cast<char*>(&vd0), 8);
149  model_file.read(reinterpret_cast<char*>(&vd1), 8);
150  model_file.read(reinterpret_cast<char*>(&vd2), 8);
151  mean_shape.at<float>(counter, 0) = static_cast<float>(vd0);
152  ++counter;
153  mean_shape.at<float>(counter, 0) = static_cast<float>(vd1);
154  ++counter;
155  mean_shape.at<float>(counter, 0) = static_cast<float>(vd2);
156  ++counter;
157  }

161  model_file.read(reinterpret_cast<char*>(&num_eigenvals_shape), 4);
162  if (num_eigenvals_shape != num_shape_pca_coeffs)
163  {
164  std::cout << "Warning: Number of coefficients in the PCA basis matrix is not equal to the number of "
165  "eigenvalues. Something will probably go wrong during the loading."
166  << std::endl;
167  }
168  Mat eigenvalues_shape(num_eigenvals_shape, 1, CV_32FC1);
169  for (unsigned int i = 0; i < num_eigenvals_shape; ++i)

273 inline std::vector<std::array<double, 2>> load_isomap(std::string isomap_file)
274 {
275  using std::string;
276  std::vector<float> x_coords, y_coords;
277  string line;
278  std::ifstream file(isomap_file);
279  if (!file)
280  {
281  const string error_msg("The isomap file could not be opened. Did you specify a correct filename? " + isomap_file);
282  throw std::runtime_error(error_msg);
283  } else
284  {
285  while (getline(file, line))
286  {
287  std::istringstream iss(line);
288  string x, y;
289  iss >> x >> y;
290  x_coords.push_back(std::stof(x));
291  y_coords.push_back(std::stof(y));
292  }
293  file.close();
294  }
295  // Process the coordinates: Find the min/max and rescale to [0, 1] x [0, 1]
296  const auto min_max_x = std::minmax_element(begin(x_coords), end(x_coords)); // min_max_x is a pair, first=min, second=max
297  const auto min_max_y = std::minmax_element(begin(y_coords), end(y_coords));
298 
299  std::vector<std::array<double, 2>> tex_coords;
300  const float divisor_x = *min_max_x.second - *min_max_x.first;
301  const float divisor_y = *min_max_y.second - *min_max_y.first;
302  for (int i = 0; i < x_coords.size(); ++i)
303  {
304  tex_coords.push_back(std::array<double, 2>{
305  (x_coords[i] - *min_max_x.first) / divisor_x,
306  1.0f - (y_coords[i] - *min_max_y.first) / divisor_y}); // We rescale to [0, 1] and at the same
307  // time flip the y-coords (because in the
308  // isomap, the coordinates are stored
309  // upside-down).
310  }
311 
312  return tex_coords;
313 };
